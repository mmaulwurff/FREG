71  return playerZ-k+'0';

184 void Screen::Print() {
185  w->ReadLock();
186 
187  if ( updated ) {
188  w->Unlock();
189  return;
190  }
191  updated=true;
192 
193  if ( player->UsingBlock() )
194  switch ( player->UsingType() ) {
195  case OPEN:
196  PrintInv(leftWin, player->UsingBlock()->HasInventory());
197  break;
198  default: PrintNormal(leftWin);
199  }
200  else
201  PrintNormal(leftWin);
202 
203  switch ( player->UsingSelfType() ) {
204  case OPEN:
205  PrintInv(rightWin, player->GetP()->HasInventory());
206  break;
207  default: PrintFront(rightWin);
208  }
209 
210  const short dur=player->HP();
211  const short breath=player->Breath();
212  const short satiation=player->Satiation();
213  w->Unlock();
214 
215  werase(hudWin);
216  ushort i;
217  //HitPoints line
218  wstandend(hudWin);
219  wprintw(hudWin, " HP: %3hd [", dur);
220  wcolor_set(hudWin, WHITE_RED, NULL);
221  for (i=0; i<10*dur/max_durability; ++i)
222  waddch(hudWin, '.');
223  wstandend(hudWin);
224  mvwaddstr(hudWin, 0, 21, "]\n");
225 
226  //breath line
227  if ( -1!=breath ) {
228  wprintw(hudWin, " BR: %3hd%% [", 100*breath/max_breath);
229  wcolor_set(hudWin, WHITE_BLUE, NULL);
230  for (i=0; i<10*breath/max_breath; ++i)
231  waddch(hudWin, '.');
232  wstandend(hudWin);
233  mvwaddstr(hudWin, 1, 21, "]\n");
234  }
235 
236  //satiation line
237  if ( -1!=satiation ) {
238  if ( seconds_in_day*time_steps_in_sec<satiation ) {
239  wcolor_set(hudWin, BLUE_BLACK, NULL);
240  mvwaddstr(hudWin, 2, 1, "Gorged\n");
241  } else if ( 3*seconds_in_day*time_steps_in_sec/4<satiation ) {
242  wcolor_set(hudWin, GREEN_BLACK, NULL);
243  mvwaddstr(hudWin, 2, 1, "Full\n");
244  } else if (seconds_in_day*time_steps_in_sec/4>satiation) {
245  wcolor_set(hudWin, RED_BLACK, NULL);
246  mvwaddstr(hudWin, 2, 1, "Hungry\n");
247  }
248  }
249 
250  wnoutrefresh(hudWin);
251  doupdate();
252 
253  //fprintf(stderr, "player x: %hu, y: %hu", player->X(), player->Y());
254 }

278  waddstr(window, " ");
279  }
280  break;
281  }
282 
283  wstandend(window);
284  box(window, 0, 0);
285  //mvwprintw(window, SCREEN_SIZE+1, 1, "Time:%d", w->TimeOfDay());
286  if ( UP==dir || DOWN==dir ) {
287  const ushort start=(shred_width*w->NumShreds()-SCREEN_SIZE)/2;
288  mvwaddstr(window, 0, 1, ( UP==dir ) ? "Sky View" : "Ground View");
289  Arrows(window , (player->X()-start)*2+1, player->Y()-start+1);

295 void Screen::PrintFront(WINDOW * const window) const {
296  const int dir=player->Dir();
297  if ( UP==dir || DOWN==dir ) {
298  wstandend(window);
299  werase(window);
300  box(window, 0, 0);
301  mvwaddstr(window, 0, 1, "No view");
302  wnoutrefresh(window);
303  return;
304  }
305 
306  short x_step, z_step,
307  x_end, z_end,
308  * x, * z,
309  i, j, k;
310  const ushort pX=player->X();
311  const ushort pY=player->Y();
312  const ushort pZ=player->Z();
313  const ushort start=(shred_width*w->NumShreds()-SCREEN_SIZE)/2;
314  ushort x_start, z_start, k_start;
315  ushort arrow_Y, arrow_X;
316  switch ( dir ) {
317  case NORTH:
318  x=&i;
319  x_step=1;
320  x_start=start;
321  x_end=start+SCREEN_SIZE;
322  z=&j;
323  z_step=-1;
324  z_start=pY-1;
325  z_end=(w->NumShreds()/2-1)*shred_width-1;
326  arrow_X=(pX-start)*2+1;
327  break;
328  case SOUTH:
329  x=&i;
330  x_step=-1;
331  x_start=SCREEN_SIZE-1+start;
332  x_end=start-1;
333  z=&j;
334  z_step=1;
335  z_start=pY+1;
336  z_end=(w->NumShreds()/2+2)*shred_width;
337  arrow_X=(SCREEN_SIZE-pX+start)*2-1;
338  break;
339  case WEST:
340  x=&j;
341  x_step=-1;
342  x_start=SCREEN_SIZE-1+start;
343  x_end=start-1;
344  z=&i;
345  z_step=-1;
346  z_start=pX-1;
347  z_end=(w->NumShreds()/2-1)*shred_width-1;
348  arrow_X=(SCREEN_SIZE-pY+start)*2-1;
349  break;
350  case EAST:
351  x=&j;
352  x_step=1;
353  x_start=start;
354  x_end=SCREEN_SIZE+start;
355  z=&i;
356  z_step=1;
357  z_start=pX+1;
358  z_end=(w->NumShreds()/2+2)*shred_width;
359  arrow_X=(pY-start)*2+1;
360  break;
361  default:
362  fprintf(stderr,
363  "Screen::PrintFront(WINDOW *): unlisted dir: %d\n",
364  (int)dir);
365  return;
366  }
367  if ( pZ+SCREEN_SIZE/2>=height ) {
368  k_start=height-2;
369  arrow_Y=height-pZ;
370  } else if ( pZ-SCREEN_SIZE/2<0 ) {
371  k_start=SCREEN_SIZE-1;
372  arrow_Y=SCREEN_SIZE-pZ;
373  } else {
374  k_start=pZ+SCREEN_SIZE/2;
375  arrow_Y=SCREEN_SIZE/2+1;
376  }
377 
378  const int block_side=w->Anti(dir);
379  wmove(window, 1, 1);
380  for (k=k_start; k_start-k<SCREEN_SIZE; --k, waddstr(window, "\n_"))
381  for (*x=x_start; *x!=x_end; *x+=x_step) {
382  for (*z=z_start; *z!=z_end; *z+=z_step)
383  if (w->Transparent(i, j, k) < 2) {
384  if ( w->Enlightened(i, j, k, block_side) &&
385  player->Visible(i, j, k) ) {
386  wcolor_set(window, Color(i, j, k), NULL);
387  waddch(window, CharName(i, j, k));
388  waddch(window, CharNumberFront(i, j));
389  } else {
390  wcolor_set(window, BLACK_BLACK, NULL);
391  waddstr(window, " ");
392  }
393  break;
394  }
395  if (*z==z_end) { // ( )
396  *z-=z_step;
397  wcolor_set(window,
398  (player->Visible(i, j, k) ? WHITE_BLUE : BLACK_BLACK),
399  NULL);
400  waddstr(window, " .");
401  }
402  }
403 
404  wstandend(window);
405  box(window, 0, 0);
406  mvwaddstr(window, 0, 1, "Front View");
407  Arrows(window, arrow_X, arrow_Y);
408  wnoutrefresh(window);
409 }
