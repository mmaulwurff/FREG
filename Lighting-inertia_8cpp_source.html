20 /**\file Lighting-inertia.cpp
21  * \brief This file provides simple (also mad) lighting for freg.
22  *
23  * It has light inertia, meaning that block, enlightened by outer
24  * source, will remain enlightened when light source is removed.
25  * Light is divided to sunlight and other light. Sunlight is
26  * changing over the day.
27  *
28  * This lighting can be used to develop more intelligent one.
29  * Intended to be fast (and is!).
30  *
31  * LightMap is uchar:
32  * & 0xF0 bits are for non-sun light,
33  * & 0x0F bits for sun. */

58 /// Makes block emit shining.
59 /** Receives only non-sun light as level, from 1 to F. */
60 void World::Shine(const int x, const int y, const int z, int level,
61  const bool init)
62 {
63  const int transparent = GetBlock(x, y, z)->Transparent();
64  if ( SetFireLightMap(level << 4, x, y, z) ) {
65  if ( INVISIBLE != transparent && not initial_lighting ) {
66  emit Updated(x, y, z);
67  }
68  if ( not init ) {
69  return;
70  }
71  }
72  if ( (transparent != BLOCK_OPAQUE && level > 1) || init ) {
73  --level;
74  const int border = GetBound();
75  if ( x > 0 ) Shine(x-1, y, z, level, false);
76  if ( x < border ) Shine(x+1, y, z, level, false);
77  if ( y > 0 ) Shine(x, y-1, z, level, false);
78  if ( y < border ) Shine(x, y+1, z, level, false);
79  Shine(x, y, z-1, level, false);
80  Shine(x, y, z+1, level, false);
81  }
82 }

84 void World::SunShineVertical(const int x, const int y, int z, int light_lev) {
85  /* 2 1 3
86  * * First, light goes down, then divides to 4 branches
87  * ^ | ^ to N-S-E-W, and goes up.
88  * | | |
89  * | | |
90  * |<v>|
91  * # */
92  Shred * const shred = GetShred(x, y);
93  const int x_in = Shred::CoordInShred(x);
94  const int y_in = Shred::CoordInShred(y);
95  shred->SetLightmap(x_in, y_in, HEIGHT-1, 0xFF);
96  for ( ; ; --z) {
97  shred->SetSunLight(x_in, y_in, z, light_lev);
98  switch ( shred->GetBlock(x_in, y_in, z)->Transparent() ) {
99  case BLOCK_OPAQUE: CrossUpShine(x, y, z); return;
100  case BLOCK_TRANSPARENT:
101  --light_lev;
102  if ( not initial_lighting ) {
103  emit Updated(x, y, z);
104  }
105  break;
106  case INVISIBLE: break;
107  }
108  }
109 }

111 void World::CrossUpShine(const int x, const int y, const int z_bottom) {
112  const int bound = GetBound();
113  if ( initial_lighting ) {
114  if ( x > 0 ) UpShineInit(x-1, y, z_bottom);
115  if ( x < bound ) UpShineInit(x+1, y, z_bottom);
116  if ( y > 0 ) UpShineInit(x, y-1, z_bottom);
117  if ( y < bound ) UpShineInit(x, y+1, z_bottom);
118  } else {
119  if ( x > 0 ) UpShine(x-1, y, z_bottom);
120  if ( x < bound ) UpShine(x+1, y, z_bottom);
121  if ( y > 0 ) UpShine(x, y-1, z_bottom);
122  if ( y < bound ) UpShine(x, y+1, z_bottom);
123  emit Updated(x, y, z_bottom);
124  }
125 }

172 void World::ReEnlightenMove(const int dir) {
173  initial_lighting = true;
174  switch ( dir ) {

188  for (int i=0; i<NumShreds(); ++i) {
189  shreds[NumShreds()*i+NumShreds()-1]->ShineAll();
190  shreds[NumShreds()*i+NumShreds()-2]->ShineAll();

194  for (int i=0; i<NumShreds(); ++i) {
195  shreds[NumShreds()*i ]->ShineAll();
196  shreds[NumShreds()*i+1]->ShineAll();

199  default: fprintf(stderr, "%s: dir (?): %d\n", Q_FUNC_INFO, dir);
200  }
