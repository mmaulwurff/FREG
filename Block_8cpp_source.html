112 void Block::Damage(const int dmg, int dmg_kind) {
113  if ( dmg_kind > DAMAGE_PUSH_UP ) {
114  dmg_kind = DAMAGE_PUSH_UP;
115  }
116  switch ( Sub() ) {
117  case SUB_DUST:
118  case GREENERY:
119  case ROSE:
120  case HAZELNUT: durability = 0; return;
121  case NULLSTONE:
122  case STAR:
123  case AIR:
124  case SKY: return;
125  }
126  switch ( dmg_kind ) {
127  case DAMAGE_ULTIMATE: durability = 0; return;
128  case DAMAGE_NO: return;
129  case DAMAGE_TIME: durability -= dmg; return;
130  case DAMAGE_ACID:
131  switch ( Sub() ) {
132  default: durability -= 2 * dmg; return;
133  case DIFFERENT: durability = 0; return;
134  case ACID:
135  case ADAMANTINE:
136  case IRON:
137  case GLASS: return;
138  }
139  }
140  int mult = 1; // default
141  switch ( Sub() ) {
142  case DIFFERENT: return;
143  case MOSS_STONE:
144  case STONE: switch ( dmg_kind ) {
145  case DAMAGE_HEAT:
146  case DAMAGE_HANDS:
147  case DAMAGE_PUSH_UP:
148  case DAMAGE_CUT: return;
149  case DAMAGE_MINE: mult = 2; break;
150  } break;
151  case WOOD: switch ( dmg_kind ) {
152  case DAMAGE_CUT: mult = 2; break;
153  case DAMAGE_PUSH_UP:
154  case DAMAGE_HANDS: return;
155  } break;
156  case A_MEAT:
157  case H_MEAT: switch ( dmg_kind ) {
158  case DAMAGE_HEAT:
159  case DAMAGE_THRUST: mult = 2; break;
160  case DAMAGE_PUSH_UP: mult = ( dmg < 10 ) ? 0 : 1;
161  } break;
162  case SAND:
163  case SOIL: mult += ( DAMAGE_DIG == dmg_kind ); break;
164  case ADAMANTINE:
165  case FIRE: mult = ( DAMAGE_FREEZE == dmg_kind ); break;
166  case WATER: mult = ( DAMAGE_HEAT == dmg_kind ); break;
167  case GLASS: switch ( dmg_kind ) {
168  case DAMAGE_PUSH_UP: durability -= dmg*((MAX_DURABILITY+9)/10); return;
169  default: durability = 0; // no break;
170  case DAMAGE_HEAT: return;
171  }
172  case IRON: switch ( dmg_kind ) {
173  case DAMAGE_HANDS:
174  case DAMAGE_PUSH_UP: return;
175  }
176  }
177  durability -= mult*dmg;
178 } // Block::Damage(const ind dmg, const int dmg_kind)
