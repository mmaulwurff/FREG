122  void Grass::DoRareAction() {
123  World * const world = GetWorld();
124  if ( FIRE == Sub() ) {
125  DamageAround();
126  if ( qrand()%10 || IsSubAround(WATER) ) {
127  Damage(2, DAMAGE_FREEZE);
128  }
129  }
130  if ( not IsBase(Sub(), world->GetBlock(X(), Y(), Z()-1)->Sub()) ) {
131  world->DestroyAndReplace(X(), Y(), Z());
132  return;
133  } // else:
134  int i=X(), j=Y(), k=Z();
135  // increase this if grass grows too fast
136  switch ( qrand() % (FIRE==Sub() ? 4 : SECONDS_IN_HOUR*2) ) {
137  case 0: ++i; break;
138  case 1: --i; break;
139  case 2: ++j; break;
140  case 3: --j; break;
141  default: return;
142  }
143  if ( not world->InBounds(i, j) ) return;
144  const int sub_near = world->GetBlock(i, j, k)->Sub();
145  if ( world->Enlightened(i, j, k) || FIRE == Sub() ) {
146  if ( AIR == sub_near
147  && IsBase(Sub(), world->GetBlock(i, j, k-1)->Sub() ) )
148  {
149  world->Build(BlockManager::NewBlock(GRASS, Sub()), i, j, k);
150  } else if ( IsBase(Sub(), sub_near)
151  && AIR == world->GetBlock(i, j, ++k)->Sub() )
152  {
153  world->Build(BlockManager::NewBlock(GRASS, Sub()), i, j, k);
154  }
155  }

386  if ( alarmTime == current_time ) {
387  Block::Inscribe(tr("Alarm. %1").arg(GetWorld()->TimeOfDayStr()));

389  } else if ( timerTime > 0 ) {

511  usage_types Map::Use(Block * const who) {
512  if ( noteId == 0 ) {
513  who->ReceiveSignal(QObject::tr("Set title to this map first."));
514  return USAGE_TYPE_NO;
515  } // else:

517  const Active * const active = who->ActiveBlock();
518  if ( active == nullptr ) return USAGE_TYPE_READ;
519  QFile map_file(home_path + active->GetWorld()->WorldName()
520  + "/texts/" + GetNote());
521  if ( not map_file.open(QIODevice::ReadWrite | QIODevice::Text) ) {
522  return USAGE_TYPE_READ;
523  }
524  const long lati = active->GetShred()->Latitude();
525  const long longi = active->GetShred()->Longitude();
526  const int FILE_SIZE_CHARS = 31 + 1;
527  if ( 0 == map_file.size() ) { // new map
528  const char header[FILE_SIZE_CHARS+1] =
529  "+-----------------------------+\n";
530  const char body[FILE_SIZE_CHARS+1] =
531  "| |\n";
532  map_file.write(header, FILE_SIZE_CHARS);
533  for (int i=0; i<FILE_SIZE_CHARS-3; ++i) {
534  map_file.write(body, FILE_SIZE_CHARS);
535  }
536  map_file.write(header, FILE_SIZE_CHARS);
537  longiStart = longi;
538  latiStart = lati;
539  }
540  const int border_dist = (FILE_SIZE_CHARS - 1) / 2;
541  if (
542  ( qAbs(lati - latiStart ) > border_dist ) ||
543  ( qAbs(longi - longiStart) > border_dist ) )
544  {
545  return USAGE_TYPE_READ;
546  }
547  if ( savedChar ) {
548  map_file.seek(savedShift);

553  lati - latiStart + border_dist );
554  map_file.putChar('@');
555  savedChar = active->GetWorld()->GetMap()->TypeOfShred(longi, lati);
556  map_file.seek(FILE_SIZE_CHARS * (FILE_SIZE_CHARS - 1));
