365 void Player::ProcessCommand(QString command) {
366  if ( command.isEmpty() ) return;
367  const QMutexLocker locker(world->GetLock());
368  QTextStream comm_stream(&command);
369  QString request;
370  comm_stream >> request;
371  if ( "give"==request || "get"==request ) {
372  if ( not (GetCreativeMode() || COMMANDS_ALWAYS_ON) ) {
373  emit Notify(tr("You are not in Creative Mode."));
374  return;
375  } // else:
376  Inventory * const inv = PlayerInventory();
377  if ( inv == nullptr ) return;
378  QString kind, sub;
379  comm_stream >> kind >> sub;
380  const int kind_code = BlockManager::StringToKind(kind);
381  if ( kind_code == LAST_KIND ) {
382  emit Notify(tr("%1 command: invalid kind!").arg(request));
383  return;
384  } // else:
385  const int sub_code = sub.isEmpty() ?
386  static_cast<int>(STONE) : BlockManager::StringToSub(sub);
387  if ( sub_code == LAST_SUB ) {
388  emit Notify(tr("%1 command: invalid substance!").arg(request));
389  return;
390  } // else:
391  emit Notify(QString("sub: %1").arg(sub_code));
392  Block * const block = block_manager.NewBlock(kind_code, sub_code);
393  if ( not inv->Get(block) ) {
394  emit Notify(tr("No place for requested thing."));
395  block_manager.DeleteBlock(block);
396  } else {
397  emit Updated();
398  }
399  } else if ( "move" == request ) {
400  int direction;
401  comm_stream >> direction;
402  Move(static_cast<dirs>(direction));
403  } else if ( "time" == request ) {
404  emit Notify( (GetCreativeMode() || COMMANDS_ALWAYS_ON) ?
405  GetWorld()->TimeOfDayStr() : tr("Not in Creative Mode.") );
406  } else if ( "version" == request ) {
407  emit Notify(tr("freg version: %1. Compiled on %2 at %3 with Qt %4.\n\
408 Current Qt version: %5. Build type: %6. Compiler: %7.").
409  arg(VER).arg(__DATE__).arg(__TIME__).arg(QT_VERSION_STR).
410  arg(qVersion()).arg(DEBUG ? tr("debug") : tr("release")).
411  arg(COMPILER));
412  } else if ( "help" == request ) {
413  comm_stream >> request;
414  emit ShowFile( QString("help_%1/%2.txt")
415  .arg(locale.left(2)).arg(request) );
416  } else {
417  emit Notify(tr("Don't know such command: \"%1\".").arg(command));
418  }
419 } // void Player::ProcessCommand(QString command)
