329 void Player::ProcessCommand(QString command) {
330  QTextStream comm_stream(&command);
331  QByteArray request;
332  comm_stream >> request;
333  const QMutexLocker locker(world->GetLock());
334  switch ( UniqueIntFromString(request.constData()) ) {
335  case UniqueIntFromString("give"):
336  case UniqueIntFromString("get" ): {
337  if ( ForbiddenAdminCommands() ) return;
338  Inventory * const inv = PlayerInventory();
339  if ( inv == nullptr ) return;
340  QByteArray kind, sub;
341  comm_stream >> kind >> sub;
342  const int kind_code = BlockManager::StringToKind(kind);
343  if ( kind_code == LAST_KIND ) {
344  emit Notify(tr("%1 command: invalid kind!").arg(QString(request)));
345  return;
346  } // else:
347  const int sub_code = sub.isEmpty() ?
348  static_cast<int>(STONE) : BlockManager::StringToSub(sub);
349  if ( sub_code == LAST_SUB ) {
350  emit Notify(tr("%1 command: invalid substance!")
351  .arg(QString(request)));
352  return;
353  } // else:
354  Block * const block = BlockManager::NewBlock(kind_code, sub_code);
355  if ( inv->Get(block) ) {
356  emit Updated();
357  } else {
358  block_manager.DeleteBlock(block);
359  }
360  } break;
361  case UniqueIntFromString("move"): {
362  int direction;
363  comm_stream >> direction;
364  Move(static_cast<dirs>(direction));
365  } break;
366  case UniqueIntFromString("time"):
367  if ( ForbiddenAdminCommands() ) return;
368  emit Notify(GetWorld()->TimeOfDayStr());
369  break;
370  case UniqueIntFromString("version"):
371  emit Notify(tr("freg version: %1. Compiled on %2 at %3 with Qt %4.")
372  .arg(VER)
373  .arg(__DATE__)
374  .arg(__TIME__)
375  .arg(QT_VERSION_STR));
376  emit Notify(tr("Current Qt version: %1. Build type: %2. Compiler: %3.")
377  .arg(qVersion())
378  .arg(DEBUG ? tr("debug") : tr("release"))
379  .arg(COMPILER));
380  break;
381  case UniqueIntFromString("warranty"): comm_stream << "warranty";
382  case UniqueIntFromString("help"):
383  comm_stream >> request;
384  if ( request.isEmpty() ) {
385  request = "help";
386  }
387  emit ShowFile( QString("help_%1/%2.md")
388  .arg(locale.left(2)).arg(QString(request)) );
389  break;
390  default:
391  emit Notify(tr("Don't know such command: \"%1\".").arg(command));
392  break;
393  }
394 } // void Player::ProcessCommand(QString command)
