370 void Player::ProcessCommand(QString command) {
371  if ( command.isEmpty() ) return;
372  const QWriteLocker locker(world->GetLock());
373  QTextStream comm_stream(&command);
374  QString request;
375  comm_stream >> request;
376  if ( "give"==request || "get"==request ) {
377  if ( not (GetCreativeMode() || COMMANDS_ALWAYS_ON) ) {
378  emit Notify(tr("You are not in Creative Mode."));
379  return;
380  } // else:
381  Inventory * const inv = PlayerInventory();
382  if ( inv == nullptr ) return;
383  QString kind, sub;
384  comm_stream >> kind >> sub;
385  const quint8 kind_code = BlockManager::StringToKind(kind);
386  if ( kind_code == LAST_KIND ) {
387  emit Notify(tr("%1 command: invalid kind!").arg(request));
388  return;
389  } // else:
390  const quint8 sub_code = sub.isEmpty() ?
391  static_cast<quint8>(STONE) : BlockManager::StringToSub(sub);
392  if ( sub_code == LAST_SUB ) {
393  emit Notify(tr("%1 command: invalid substance!").arg(request));
394  return;
395  } // else:
396  emit Notify(QString("sub: %1").arg(sub_code));
397  Block * const block = block_manager.NewBlock(kind_code, sub_code);
398  if ( not inv->Get(block) ) {
399  emit Notify(tr("No place for requested thing."));
400  block_manager.DeleteBlock(block);
401  } else {
402  emit Updated();
403  }
404  } else if ( "move" == request ) {
405  int direction;
406  comm_stream >> direction;
407  Move(direction);
408  } else if ( "what" == request ) {
409  ushort x_what, y_what, z_what;
410  comm_stream >> x_what >> y_what >> z_what;
411  if ( not world->InBounds(x_what, y_what, z_what) ) {
412  emit Notify(tr("Such block is out of loaded world."));
413  } else if ( GetCreativeMode() || COMMANDS_ALWAYS_ON
414  || qAbs(x-x_what) > 1
415  || qAbs(y-y_what) > 1
416  || qAbs(z-z_what) > 1 )
417  {
418  Examine(x_what, y_what, z_what);
419  } else {
420  emit Notify(tr("Too far."));
421  }
422  } else if ( "time" == request ) {
423  emit Notify( (GetCreativeMode() || COMMANDS_ALWAYS_ON) ?
424  GetWorld()->TimeOfDayStr() :
425  tr("Not in Creative Mode.") );
426  } else if ( "version" == request ) {
427  emit Notify(tr("freg version: %1. Compiled on %2 at %3.").
428  arg(VER).arg(__DATE__).arg(__TIME__));
429  } else if ( "help" == request ) {
430  comm_stream >> request;
431  emit ShowFile( QString("help_%1/%2.txt")
432  .arg(locale.left(2)).arg(request));
433  } else {
434  emit Notify(tr("Don't know such command: \"%1\".").arg(command));
435  }
436 } // void Player::ProcessCommand(QString command)
