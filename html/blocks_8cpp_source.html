70  int Block::Damage(const ushort dmg, const int dmg_kind) {
71  ushort mult=1;
72  switch ( Sub() ) {
73  case DIFFERENT:
74  if ( TIME==dmg_kind ) {
75  return durability=0;
76  }
77  //no break
78  case NULLSTONE:
79  case STAR:
80  case AIR:
81  case SKY:
82  case SUN_MOON: mult=0; break;
83  case WATER:
84  mult=( HEAT==dmg_kind || TIME==dmg_kind );
85  break;
86  case MOSS_STONE:
87  case STONE:
88  switch ( dmg_kind ) {
89  case CRUSH: mult=0; break;
90  case MINE: mult=2; break;
91  }
92  break;
93  case GREENERY:
94  case GLASS: return durability=0;
95  case ROSE:
96  case HAZELNUT:
97  case WOOD: mult=1+(CUT==dmg_kind); break;
98  case SAND:
99  case SOIL: mult=1+(DIG==dmg_kind); break;
100  case A_MEAT:
101  case H_MEAT: mult=1+(THRUST==dmg_kind); break;
102  }
103  return durability-=mult*dmg;
104  } //Block::Damage
