229  void Grass::DoRareAction() {
230  World * const world = GetWorld();
231  if ( FIRE == Sub() ) {
232  const Xyz coords[] = {
233  Xyz( X()-1, Y(), Z() ),
234  Xyz( X()+1, Y(), Z() ),
235  Xyz( X(), Y()-1, Z() ),
236  Xyz( X(), Y()+1, Z() ),
237  Xyz( X(), Y(), Z()-1 ),
238  Xyz( X(), Y(), Z()+1 )
239  };
240  for (const Xyz xyz : coords) {
241  if ( world->InBounds(xyz.X(), xyz.Y()) ) {
242  TryDestroy(xyz.X(), xyz.Y(), xyz.Z());
243  }
244  }
245  if ( qrand()%10 || IsSubAround(WATER) ) {
246  Damage(2, FREEZE);
247  }
248  }
249  if ( not IsBase(Sub(), world->GetBlock(X(), Y(), Z()-1)->Sub()) ) {
250  world->DestroyAndReplace(X(), Y(), Z());
251  return;
252  } // else:
253  int i=X(), j=Y();
254  // increase this if grass grows too fast
255  switch ( qrand() % (FIRE==Sub() ? 4 : SECONDS_IN_HOUR*2) ) {
256  case 0: ++i; break;
257  case 1: --i; break;
258  case 2: ++j; break;
259  case 3: --j; break;
260  default: return;
261  }
262  int k = Z();
263  if ( not world->InBounds(i, j) ) return;
264  const int sub_near = world->GetBlock(i, j, k)->Sub();
265  if ( world->Enlightened(i, j, k) || FIRE == Sub() ) {
266  if ( AIR == sub_near
267  && IsBase(Sub(), world->GetBlock(i, j, k-1)->Sub() ) )
268  {
269  world->Build(block_manager.NewBlock(GRASS, Sub()), i, j, k);
270  } else if ( IsBase(Sub(), sub_near)
271  && AIR == world->GetBlock(i, j, ++k)->Sub() )
272  {
273  world->Build(block_manager.NewBlock(GRASS, Sub()), i, j, k);
274  }
275  }
276  }
