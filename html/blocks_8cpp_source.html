217  void Grass::DoRareAction() {
218  World * const world = GetWorld();
219  if ( FIRE == Sub() ) {
220  const Xyz coords[] = {
221  Xyz( X()-1, Y(), Z() ),
222  Xyz( X()+1, Y(), Z() ),
223  Xyz( X(), Y()-1, Z() ),
224  Xyz( X(), Y()+1, Z() ),
225  Xyz( X(), Y(), Z()-1 ),
226  Xyz( X(), Y(), Z()+1 )
227  };
228  for (const Xyz xyz : coords) {
229  if ( world->InBounds(xyz.X(), xyz.Y()) ) {
230  TryDestroy(xyz.X(), xyz.Y(), xyz.Z());
231  }
232  }
233  if ( qrand()%10 || IsSubAround(WATER) ) {
234  Damage(2, FREEZE);
235  }
236  }
237  if ( not IsBase(Sub(), world->GetBlock(X(), Y(), Z()-1)->Sub()) ) {
238  world->DestroyAndReplace(X(), Y(), Z());
239  return;
240  } // else:
241  int i=X(), j=Y();
242  // increase this if grass grows too fast
243  switch ( qrand() % (FIRE==Sub() ? 4 : SECONDS_IN_HOUR*2) ) {
244  case 0: ++i; break;
245  case 1: --i; break;
246  case 2: ++j; break;
247  case 3: --j; break;
248  default: return;
249  }
250  int k = Z();
251  if ( not world->InBounds(i, j) ) return;
252  const int sub_near = world->GetBlock(i, j, k)->Sub();
253  if ( world->Enlightened(i, j, k) || FIRE == Sub() ) {
254  if ( AIR == sub_near
255  && IsBase(Sub(), world->GetBlock(i, j, k-1)->Sub() ) )
256  {
257  world->Build(block_manager.NewBlock(GRASS, Sub()), i, j, k);
258  } else if ( IsBase(Sub(), sub_near)
259  && AIR == world->GetBlock(i, j, ++k)->Sub() )
260  {
261  world->Build(block_manager.NewBlock(GRASS, Sub()), i, j, k);
262  }
263  }
264  }

623  usage_types Map::Use(Block * const who) {
624  if ( note == nullptr ) {
625  if ( who ) {
626  who->ReceiveSignal(QObject::tr(
627  "Set title to this map first."));
628  }
629  return USAGE_TYPE_NO;
630  } else if ( who && who->ActiveBlock() ) {
631  const Active * const active = who->ActiveBlock();
632  QFile map_file(active->GetWorld()->
633  WorldName() + "/texts/" + *note + ".txt");
634  if ( not map_file.open(QIODevice::ReadWrite | QIODevice::Text) ) {
635  return USAGE_TYPE_READ;
636  }
637  const Shred * const shred = active->GetShred();
638  const long lati = shred->Latitude();
639  const long longi = shred->Longitude();
640  static const int FILE_SIZE_CHARS = 31;
641  if ( 0 == map_file.size() ) { // new map
642  char header[FILE_SIZE_CHARS+1];
643  memset(header, '-', FILE_SIZE_CHARS);
644  header[0] = header[FILE_SIZE_CHARS/2] =
645  header[FILE_SIZE_CHARS-1] = '+';
646  char body[FILE_SIZE_CHARS+1];
647  memset(body, ' ', FILE_SIZE_CHARS);
648  body[0] = body[FILE_SIZE_CHARS-1] = '|';
649  body[FILE_SIZE_CHARS] = header[FILE_SIZE_CHARS] = '\n';
650  map_file.write(header, FILE_SIZE_CHARS+1);
651  for (int i=0; i<FILE_SIZE_CHARS-2; ++i) {
652  map_file.write(body,FILE_SIZE_CHARS+1);
653  }
654  map_file.write(header, FILE_SIZE_CHARS+1);
655 
656  map_file.seek( FILE_SIZE_CHARS/2 * (FILE_SIZE_CHARS+1) );
657  map_file.putChar('+');
658  map_file.seek(FILE_SIZE_CHARS/2*
659  (FILE_SIZE_CHARS+1)+FILE_SIZE_CHARS-1);
660  map_file.putChar('+');
661 
662  longiStart = longi;
663  latiStart = lati;
664  }
665  if ( ( qAbs(lati-latiStart) > FILE_SIZE_CHARS/2 )
666  || ( qAbs(longi-longiStart) > FILE_SIZE_CHARS/2 ) )
667  {
