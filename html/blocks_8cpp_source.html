69  void Block::Damage(const ushort dmg, const int dmg_kind) {
70  if ( dmg_kind == NO_HARM ) return;
71  ushort mult = 1; // default
72  switch ( Sub() ) {
73  case DIFFERENT:
74  if ( TIME == dmg_kind ) {
75  durability = 0;
76  return;
77  }
78  // no break
79  case NULLSTONE:
80  case STAR:
81  case AIR:
82  case SKY:
83  case SUN_MOON: return;
84  case WATER: mult = ( HEAT==dmg_kind || TIME==dmg_kind ); break;
85  case MOSS_STONE:
86  case STONE: switch ( dmg_kind ) {
87  case CRUSH: case DAMAGE_HANDS:
88  case CUT: return;
89  case MINE: mult = 2; break;
90  } break;
91  case GREENERY:
92  case GLASS: durability = 0; return;
93  case WOOD: switch ( dmg_kind ) {
94  default: mult = 1; break;
95  case CUT: mult = 2; break;
96  case DAMAGE_HANDS: return;
97  } break;
98  case SAND:
99  case A_MEAT:
100  case H_MEAT: ++(mult = (THRUST==dmg_kind)); break;
101  case SOIL: ++(mult = (DIG ==dmg_kind)); break;
102  case FIRE: mult = (FREEZE==dmg_kind || TIME==dmg_kind); break;
103  }
104  durability -= mult*dmg;
105  }
