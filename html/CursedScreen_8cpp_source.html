230 void Screen::ControlPlayer(const int ch) {
231  CleanFileToShow();
232  // Q, ctrl-c, ctrl-d, ctrl-q, ctrl-x
233  // TODO: ctrl-z (to background) support
234  if ( 'Q'==ch || 3==ch || 4==ch || 17==ch || 24==ch ) {
235  emit ExitReceived();
236  return;
237  } // else:
238  if ( ch>='a' && ch<='z' ) { // actions with inventory
239  InventoryAction(ch-'a');
240  return;
241  } // else:
242  switch ( ch ) { // interactions with world
243  default:
244  Notify(tr("Unknown key. Press 'H' for help."));
245  #ifndef QT_NO_DEBUG
246  Notify(QString("Pressed key code: %1.").arg(ch));
247  #endif
248  break;
249  case KEY_UP: case '8': MovePlayer(NORTH); break;
250  case KEY_DOWN: case '2': MovePlayer(SOUTH); break;
251  case KEY_RIGHT: case '6': MovePlayer(EAST); break;
252  case KEY_LEFT: case '4': MovePlayer(WEST); break;
253  case KEY_END: case '5': player->Move(DOWN); break;
254  case '7': MovePlayerDiag(NORTH, WEST); break;
255  case '9': MovePlayerDiag(NORTH, EAST); break;
256  case '1': MovePlayerDiag(SOUTH, WEST); break;
257  case '3': MovePlayerDiag(SOUTH, EAST); break;
258  case ' ': player->Jump(); break;
259 
260  case '>': player->SetDir(World::TurnRight(player->GetDir())); break;
261  case '<': player->SetDir(World::TurnLeft (player->GetDir())); break;
262  case KEY_NPAGE: player->SetDir(DOWN); break;
263  case KEY_PPAGE: player->SetDir(UP); break;
264 
265  case 'I':
266  case KEY_HOME: player->Backpack(); break;
267  case 8:
268  case KEY_DC: // delete
269  case KEY_BACKSPACE: player->Damage(); break;
270  case 13:
271  case 'E':
272  case '\n': player->Use(); break;
273  case '?': player->Examine(); break;
274  case '~': player->Inscribe(); break;
275  case 27: /* esc */ player->StopUseAll(); break;
276 
277  case KEY_IC: player->Build(1); break; // insert
278  case 'B': SetActionMode(ACTION_BUILD); break;
279  case 'C': SetActionMode(ACTION_CRAFT); break;
280  case 'D':
281  case 'T': SetActionMode(ACTION_THROW); break;
282  case 'N': SetActionMode(ACTION_INSCRIBE); break;
283  case 'G':
284  case 'O': SetActionMode(ACTION_OBTAIN); break;
285  case 'F':
286  case 'W': SetActionMode(ACTION_WIELD); break;
287  case 'U': SetActionMode(ACTION_USE); break;
288  case 'S':
289  if ( player->PlayerInventory() ) {
290  player->PlayerInventory()->Shake();
291  }
292  break;
293  case KEY_HELP:
294  case 'H':
295  DisplayFile(QString("help_%1/help.txt").arg(locale.left(2)));
296  break;
297  case 'R':
298  case 'L': RePrint(); break;
299 
300  case '-': shiftFocus = -!shiftFocus; break; // move focus down
301  case '+': shiftFocus = !shiftFocus; break; // move focus up
302 
303  case '!': player->SetCreativeMode( not player->GetCreativeMode() ); break;
304  case ':':
305  case '/': PassString(previousCommand); // no break
306  case '.': ProcessCommand(previousCommand); break;
307  }
308  updated = false;
309 } // void Screen::ControlPlayer(int ch)
