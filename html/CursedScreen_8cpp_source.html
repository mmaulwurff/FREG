191 void Screen::ControlPlayer(const int ch) {
192  CleanFileToShow();
193  // Q, ctrl-c, ctrl-d, ctrl-q, ctrl-x
194  // TODO: ctrl-z (to background) support
195  if ( 'Q'==ch || 3==ch || 4==ch || 17==ch || 24==ch ) {
196  emit ExitReceived();
197  return;
198  } // else:
199  if ( ch>='a' && ch<='z' ) { // actions with inventory
200  InventoryAction(ch-'a');
201  return;
202  } // else:
203  switch ( ch ) { // interactions with world
204  case KEY_UP: case '8': MovePlayer(NORTH); break;
205  case KEY_DOWN: case '2': MovePlayer(SOUTH); break;
206  case KEY_RIGHT: case '6': MovePlayer(EAST); break;
207  case KEY_LEFT: case '4': MovePlayer(WEST); break;
208  case KEY_END: case '5': player->Move(DOWN); break;
209  case '7': MovePlayerDiag(NORTH, WEST); break;
210  case '9': MovePlayerDiag(NORTH, EAST); break;
211  case '1': MovePlayerDiag(SOUTH, WEST); break;
212  case '3': MovePlayerDiag(SOUTH, EAST); break;
213  case ' ': player->Jump(); break;
214  case '=': player->Move(); break;
215 
216  case '>': player->SetDir(World::TurnRight(player->GetDir())); break;
217  case '<': player->SetDir(World::TurnLeft (player->GetDir())); break;
218  case KEY_NPAGE: player->SetDir(DOWN); break;
219  case KEY_PPAGE: player->SetDir(UP); break;
220 
221  case 'I':
222  case KEY_HOME: player->Backpack(); break;
223  case 8:
224  case KEY_BACKSPACE: player->Damage(); break;
225  case 13:
226  case '\n': player->Use(); break;
227  case '?': player->Examine(); break;
228  case '~': player->Inscribe(); break;
229  case 27: /* esc */ player->StopUseAll(); break;
230 
231  case 'B': SetActionMode(ACTION_BUILD); break;
232  case 'C': SetActionMode(ACTION_CRAFT); break;
233  case 'D':
234  case 'T': SetActionMode(ACTION_THROW); break;
235  case 'E': SetActionMode(ACTION_EAT); break;
236  case 'F': SetActionMode(ACTION_TAKEOFF); break;
237  case 'N': SetActionMode(ACTION_INSCRIBE); break;
238  case 'G':
239  case 'O': SetActionMode(ACTION_OBTAIN); break;
240  case 'U': SetActionMode(ACTION_USE); break;
241  case 'W': SetActionMode(ACTION_WIELD); break;
242  case 'S':
243  if ( player->PlayerInventory() ) {
244  player->PlayerInventory()->Shake();
245  }
246  break;
247  case KEY_HELP:
248  case 'H':
249  DisplayFile(QString("help_%1/help.txt").arg(locale.left(2)));
250  break;
251  case 'L': RePrint(); break;
252  case 'R': // switch active hand
253  if ( not player->GetCreativeMode() ) {
254  player->SetActiveHand(not player->IsRightActiveHand());
255  Notify(tr("Now %1 hand is active.").
256  arg(player->IsRightActiveHand() ?
257  tr("right") : tr("left")));
258  }
259  break;
260 
261  case '-': shiftFocus = -!shiftFocus; break; // move focus down
262  case '+': shiftFocus = !shiftFocus; break; // move focus up
263 
264  case '!': player->SetCreativeMode( not player->GetCreativeMode() ); break;
265  case ':':
266  case '/': PassString(command); // no break
267  case '.': ProcessCommand(command); break;
268 
269  default: Notify(tr("Unknown key. Press 'H' for help."));
270  }
271  mutex->lock();
272  updated = false;
273  mutex->unlock();
274 } // void Screen::ControlPlayer(int ch)
