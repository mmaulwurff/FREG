252 void Screen::ControlPlayer(const int ch) {
253  CleanFileToShow();
254  // Q, ctrl-c, ctrl-d, ctrl-q, ctrl-x
255  // TODO: ctrl-z (to background) support
256  if ( 'Q'==ch || 3==ch || 4==ch || 17==ch || 24==ch ) {
257  emit ExitReceived();
258  return;
259  } // else:
260  if ( ch>='a' && ch<='z' ) { // actions with inventory
261  InventoryAction(ch-'a');
262  return;
263  } // else:
264  static bool step_trigger = true;
265  switch ( ch ) { // interactions with world
266  case KEY_UP:
267  case '8': MovePlayer(NORTH); break;
268  case KEY_DOWN:
269  case '2': MovePlayer(SOUTH); break;
270  case KEY_RIGHT:
271  case '6': MovePlayer(EAST); break;
272  case KEY_LEFT:
273  case '4': MovePlayer(WEST); break;
274  case KEY_END:
275  case '5': player->Move(DOWN); break;
276  case '7':
277  player->SetDir(NORTH);
278  player->Move(step_trigger ? NORTH : WEST);
279  step_trigger = !step_trigger;
280  break;
281  case '9':
282  player->SetDir(NORTH);
283  player->Move(step_trigger ? NORTH : EAST);
284  step_trigger = !step_trigger;
285  break;
286  case '1':
287  player->SetDir(SOUTH);
288  player->Move(step_trigger ? SOUTH : WEST);
289  step_trigger = !step_trigger;
290  break;
291  case '3':
292  player->SetDir(SOUTH);
293  player->Move(step_trigger ? SOUTH : EAST);
294  step_trigger = !step_trigger;
295  break;
296  case ' ': player->Jump(); break;
297  case '=': player->Move(); break;
298 
299  case '>': player->SetDir(World::TurnRight(player->GetDir())); break;
300  case '<': player->SetDir(World::TurnLeft (player->GetDir())); break;
301  case KEY_NPAGE: player->SetDir(DOWN); break;
302  case KEY_PPAGE: player->SetDir(UP); break;
303 
304  case KEY_HOME: player->Backpack(); break;
305  case 8:
306  case KEY_BACKSPACE: { // damage
307  ushort x, y, z;
308  ActionXyz(x, y, z);
309  player->Damage(x, y, z);
310  } break;
311  case 13:
312  case '\n': { // use
313  ushort x, y, z;
314  ActionXyz(x, y, z);
315  player->Use(x, y, z);
316  } break;
317  case '?': { // examine
318  ushort x, y, z;
319  ActionXyz(x, y, z);
320  player->Examine(x, y, z);
321  } break;
322  case '~': { // inscribe
323  ushort x, y, z;
324  ActionXyz(x, y, z);
325  player->Inscribe(x, y, z);
326  } break;
327  case 27: /* esc */ player->StopUseAll(); break;
328 
329  case 'B': SetActionMode(ACTION_BUILD); break;
330  case 'C': SetActionMode(ACTION_CRAFT); break;
331  case 'D':
332  case 'T': SetActionMode(ACTION_THROW); break;
333  case 'E': SetActionMode(ACTION_EAT); break;
334  case 'F': SetActionMode(ACTION_TAKEOFF); break;
335  case 'I': SetActionMode(ACTION_INSCRIBE); break;
336  case 'G':
337  case 'O': SetActionMode(ACTION_OBTAIN); break;
338  case 'U': SetActionMode(ACTION_USE); break;
339  case 'W': SetActionMode(ACTION_WIELD); break;
340  case KEY_HELP:
341  case 'H':
342  DisplayFile(QString("help_%1/help.txt")
343  .arg(locale.left(2)));
344  break;
345  case 'L': RePrint(); break;
346  case 'R': // switch active hand
347  if ( not player->GetCreativeMode() ) {
348  player->SetActiveHand(not player->IsRightActiveHand());
349  Notify(tr("Now %1 hand is active.").
350  arg(player->IsRightActiveHand() ?
351  tr("right") : tr("left")));
352  }
353  break;
354 
355  case '-': shiftFocus = -!shiftFocus; break; // move focus down
356  case '+': shiftFocus = !shiftFocus; break; // move focus up
357 
358  case '!': player->SetCreativeMode( not player->GetCreativeMode() ); break;
359  case ':':
360  case '/': PassString(command); // no break
361  case '.': ProcessCommand(command); break;
362 
363  default: Notify(tr("Unknown key. Press 'H' for help."));
364  }
365  mutex->lock();
366  updated = false;
367  mutex->unlock();
368 } // void Screen::ControlPlayer(int ch)
