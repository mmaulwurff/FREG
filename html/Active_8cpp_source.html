98 bool Active::Move(const int dir) {
99  switch ( dir ) {
100  case NORTH: --y_self; break;
101  case SOUTH: ++y_self; break;
102  case EAST: ++x_self; break;
103  case WEST: --x_self; break;
104  case UP: ++z_self; break;
105  }
106  bool overstep;
107  if ( DOWN == dir ) {
108  --z_self;
109  ++fall_height;
110  overstep = false;
111  } else {
112  Shred * const new_shred = GetWorld()->GetShred(X(), Y());
113  if ( (overstep = ( shred != new_shred )) ) {
114  falling = false;
115  shred->UnregisterLater(this);
116  new_shred->Register(this);
117  }
118  }
119  emit Moved(dir);
120  return overstep;
121 }

210 bool Active::Gravitate(const ushort range, const ushort down, const ushort up,
211  const ushort calmness)
212 {
213  World * const world = GetWorld();
214  // analyse world around
215  short for_north = 0, for_west = 0;
216  const ushort y_start = Y()-range;
217  const ushort z_start = Z()-down;
218  const ushort x_end = X()+range;
219  const ushort y_end = Y()+range;
220  const ushort z_end = Z()+up;
221  for (ushort x=X()-range; x<=x_end; ++x)
222  for (ushort y=y_start; y<=y_end; ++y) {
223  if ( not world->InBounds(x, y) ) {
224  continue;
225  }
226  Shred * const shred = world->GetShred(x, y);
227  const ushort x_in = Shred::CoordInShred(x);
228  const ushort y_in = Shred::CoordInShred(y);
229  for (ushort z=z_start; z<=z_end; ++z) {
230  short attractive;
231  if ( World::InVertBounds(z)
232  && ( attractive = Attractive(shred->
233  GetBlock(x_in, y_in, z)->Sub()) )
234  && world->DirectlyVisible(X(), Y(), Z(), x, y, z) )
235  {
236  if ( y!=Y() ) for_north += attractive/(Y()-y);
237  if ( x!=X() ) for_west += attractive/(X()-x);
238  }
239  }
240  }
241  // make direction and move there
242  if ( qAbs(for_north)>calmness || qAbs(for_west)>calmness ) {
243  SetDir( ( qAbs(for_north)>qAbs(for_west) ) ?
244  ( ( for_north>0 ) ? NORTH : SOUTH ) :
245  ( ( for_west >0 ) ? WEST : EAST ) );
246  return true;
247  } else {
248  return false;
249  }
250 }
