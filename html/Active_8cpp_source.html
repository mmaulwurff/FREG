162 bool Active::Gravitate(const int range, int bottom, int top,
163  const int calmness)
164 {
165  World * const world = GetWorld();
166  const int bound = world->GetBound();
167  // analyse world around
168  int for_north = 0, for_west = 0;
169  const int y_start = qMax(Y()-range, 0);
170  const int y_end = qMin(Y()+range, bound);
171  const int x_end = qMin(X()+range, bound);
172  bottom = qMax(Z()-bottom, 0);
173  top = qMin(Z()+top, HEIGHT-1);
174  for (int x=qMax(X()-range, 0); x<=x_end; ++x)
175  for (int y=y_start; y<=y_end; ++y) {
176  Shred * const shred = world->GetShred(x, y);
177  const int x_in = Shred::CoordInShred(x);
178  const int y_in = Shred::CoordInShred(y);
179  for (int z=bottom; z<=top; ++z) {
180  const int attractive =
181  Attractive(shred->GetBlock(x_in, y_in, z)->Sub());
182  if ( attractive != 0
183  && world->DirectlyVisible(X(), Y(), Z(), x, y, z) )
184  {
185  if ( y!=Y() ) for_north += attractive/(Y()-y);
186  if ( x!=X() ) for_west += attractive/(X()-x);
187  }
188  }
189  }
190  // make direction and move there
191  if ( qAbs(for_north)>calmness || qAbs(for_west)>calmness ) {
192  SetDir( ( qAbs(for_north)>qAbs(for_west) ) ?
193  ( ( for_north>0 ) ? NORTH : SOUTH ) :
194  ( ( for_west >0 ) ? WEST : EAST ) );
195  return true;
196  } else {
197  return false;
198  }
199 }
