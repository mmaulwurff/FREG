127 void Player::Examine(const short i, const short j, const short k) const {
128  world->ReadLock();
129 
130  emit Notify("------");
131  const Block * const block=world->GetBlock(i, j, k);
132  const int sub=block->Sub();
133  emit Notify( block->FullName() );
134  if ( creativeMode ) { //know more
135  emit Notify(tr(
136  "Light:%1, fire:%2, sun:%3. Transp:%4. Norm:%5. Dir:%6.").
137  arg(world->Enlightened(i, j, k)).
138  arg(world->FireLight(i, j, k)/16).
139  arg(world->SunLight(i, j, k)).
140  arg(block->Transparent()).
141  arg(block==block_manager.NormalBlock(sub)).
142  arg(block->GetDir()));
143  }
144  if ( AIR==sub || SKY==sub || SUN_MOON==sub ) {
145  world->Unlock();
146  return;
147  }
148  QString str;
149  if ( ""!=(str=block->GetNote()) ) {
150  emit Notify(tr("Inscription: ")+str);
151  }
152  emit Notify(tr("Temperature: %1.").arg(world->Temperature(i, j, k)));
153  emit Notify(tr("Durability: %1.").arg(block->Durability()));
154  emit Notify(tr("Weight: %1.").arg(block->Weight()));
155  world->Unlock();
156 }

353 void Player::ProcessCommand(QString & command) {
354  //don't forget lock and unlock world mutex where needed:
355  //world->WriteLock();
356  //or
357  //world->ReadLock();
358  //world->Unlock();
359  QTextStream comm_stream(&command);
360  QString request;
361  comm_stream >> request;
362  if ( "give"==request || "get"==request ) {
363  world->WriteLock();
364  Inventory * const inv=PlayerInventory();
365  if ( !creativeMode ) {
366  emit Notify(tr("You are not in Creative Mode."));
367  } else if ( inv ) {
368  int kind, sub, num;
369  comm_stream >> kind >> sub >> num;
370  if ( num <= 0 ) {
371  num=1;
372  }
373  while ( num && inv->HasRoom() ) {
374  for (ushort i=9; i && num; --i) {
375  inv->Get(block_manager.
376  NewBlock(kind, sub));
377  --num;
378  }
379  }
380  if ( num > 0 ) {
381  emit Notify(tr("No place for %1 things.").
382  arg(num));
383  }
384  emit Updated();
385  } else {
386  emit Notify(tr("No room."));
387  }
388  world->Unlock();
389  } else if ( "move"==request ) {
390  int dir;
391  comm_stream >> dir;
392  Move(dir);
393  } else if ( "what"==request ) {
394  ushort x_what, y_what, z_what;
395  comm_stream >> x_what >> y_what >> z_what;
396  world->ReadLock();
397  if ( creativeMode ) {
398  Examine(x_what, y_what, z_what);
399  } else {
400  if (
401  qAbs(x-x_what) > 1 ||
402  qAbs(y-y_what) > 1 ||
403  qAbs(z-z_what) > 1)
404  {
405  emit Notify(tr("Too far."));
406  } else {
407  Examine(x_what, y_what, z_what);
408  }
409  }
410  world->Unlock();
411  } else if ( "moo"==request ) {
412  emit Notify("^__^");
413  emit Notify("(oo)\\_______");
414  emit Notify("(__)\\ )\\/\\");
415  emit Notify(" ||----w |");
416  emit Notify(" || ||");
417  } else {
418  emit Notify(tr("Don't know such command: \"%1\".").
419  arg(command));
420  }
421 } //Player::ProcessCommand

465 void Player::CheckOverstep(const int dir) {
466  UpdateXYZ();
467  if (
468  x < (world->NumShreds()/2-1)*SHRED_WIDTH ||
469  y < (world->NumShreds()/2-1)*SHRED_WIDTH ||
470  x >= (world->NumShreds()/2+2)*SHRED_WIDTH ||
471  y >= (world->NumShreds()/2+2)*SHRED_WIDTH )
472  {
473  emit OverstepBorder(dir);
474  if ( GetCreativeMode() ) {
475  //coordinates of normal (non-creative) player are
476  //reloaded (shifted corresponding to world reload)
477  //automatically since such player is registered in
478  //his shred.
479  //This helps creative player shift his coordinates.
480  switch ( dir ) {
481  case NORTH: player->ReloadToNorth(); break;
482  case SOUTH: player->ReloadToSouth(); break;
483  case EAST: player->ReloadToEast(); break;
484  case WEST: player->ReloadToWest(); break;
485  }
486  }
487  UpdateXYZ();
488  }
489  emit Moved(GlobalX(), GlobalY(), z);
490  //for curses screen to update itself in creative mode
491  //IDEA: make signal to tell srceen player coordinates in creative mode
492  emit Updated();
493 }
