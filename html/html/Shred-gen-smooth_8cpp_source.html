38 float Noise2(const int x, const int y) { //range - [-1;1]
39  int n = x + y * 57;
40  n = (n << 13) ^ n;
41  return ((n * (n * n * 15731 + 789221) + 1376312589) & 0x7fffffff) /
42  1073741823.5f - 1.0f;
43 }

45 float InterpolatedNoise(const short x, const short y) { //range - [-1;1]
46  const short X = x >> 6;
47  const short Y = y >> 6;
48 
49  const float noise[4]={
50  Noise2(X, Y),
51  Noise2(X + 1, Y),
52  Noise2(X + 1, Y + 1),
53  Noise2(X, Y + 1)
54  };
55 
56  const float dx = (float (x) - float (X << 6))*0.015625f;
57  const float dy = (float (y) - float (Y << 6))*0.015625f;
58 
59  const float interp_x[2]={
60  dy * noise[3] + (1.0f - dy) * noise[0],
61  dy * noise[2] + (1.0f - dy) * noise[1]
62  };
63  return interp_x[1] * dx + interp_x[0] * (1.0f - dx);
64 }

66 float FinalNoise(short x, short y) { //range [-1;1] middle= 0
67  float r = 0.5f * InterpolatedNoise(x, y);
68 
69  x <<= 1, y <<= 1;
70  r += 0.25f * InterpolatedNoise(x, y);
71 
72  x <<= 1, y <<= 1;
73  return (r += 0.125f * InterpolatedNoise(x, y));
74 }

106 void Shred::ShredLandAmplitudeAndLevel(const long longi, const long lati,
107  ushort * const l, float * const a)
108 const {
109  const char shred_types[8]={
110  TypeOfShred(longi + 1, lati),
111  TypeOfShred(longi - 1, lati),
112  TypeOfShred(longi, lati + 1),
113  TypeOfShred(longi, lati - 1),
114 
115  TypeOfShred(longi + 1, lati + 1),
116  TypeOfShred(longi - 1, lati - 1),
117  TypeOfShred(longi - 1, lati + 1),
118  TypeOfShred(longi + 1, lati - 1)
119  };
120  float a2;
121  ushort l2;
122 
123  const char this_shred_type = TypeOfShred(longi, lati);
124  ShredNominalAmplitudeAndLevel(this_shred_type, &l2, &a2);
125  float amplitude, level;
126  const float this_shred_amplitude = amplitude = a2;
127  const float this_shred_level = level = float (l2);
128  for (int i = 0; i < 8; ++i) {
129  if (shred_types[i] != this_shred_type) {
130  ShredNominalAmplitudeAndLevel(shred_types[i], &l2,&a2);
131  amplitude += (a2 - this_shred_amplitude) * (1.0f/9.0f);
132  level += (float (l2) - this_shred_level)*(1.0f / 9.0f);
133  }
134  }
135  *a = amplitude;
136  *l = ushort(level);
137 }

177 void Shred::NormalUnderground(const ushort depth, const int sub) {
178  float amplitude[9];
179  ushort level[9];
180  const struct {
181  ushort * lev;
182  float * ampl;
183  } ampllev[]={
184  /* 7 1 6
185  * 3 8 2
186  * 4 0 5 */
187  {&level[7], &amplitude[7]},
188  {&level[1], &amplitude[1]},
189  {&level[6], &amplitude[6]},
190  {&level[3], &amplitude[3]},
191  {&level[8], &amplitude[8]},
192  {&level[2], &amplitude[2]},
193  {&level[4], &amplitude[4]},
194  {&level[0], &amplitude[0]},
195  {&level[5], &amplitude[5]},
196  };
197  for (int i=-1; i<=1; ++i)
198  for (int j=-1; j<=1; ++j) {
199  ShredLandAmplitudeAndLevel(i+longitude, j+latitude,
200  ampllev[(i+1)*3+j+1].lev,
201  ampllev[(i+1)*3+j+1].ampl);
202  }
203  for (ushort i = 0; i < SHRED_WIDTH; ++i)
204  for (ushort j = 0; j < SHRED_WIDTH; ++j) {
205  /* 3+---+2
206  * |...|
207  * |...|
208  * 0+---+1 */
209  //interpolate land amplitude and level
210  float il[4], ia[4];
211  float ik_x, ik_y;
212  if (j < SHRED_WIDTH / 2 && i < SHRED_WIDTH / 2) {
213  il[0] = float (level[3]), ia[0] = amplitude[3];
214  il[1] = float (level[8]), ia[1] = amplitude[8];
215  il[2] = float (level[1]), ia[2] = amplitude[1];
216  il[3] = float (level[7]), ia[3] = amplitude[7];
217  ik_x = float (i + SHRED_WIDTH / 2) * 0.0625f;
218  ik_y = float (j + SHRED_WIDTH / 2) * 0.0625f;
219  } else if (j < SHRED_WIDTH / 2 && i >= SHRED_WIDTH / 2) {
220  il[0] = float (level[8]), ia[0] = amplitude[8];
221  il[1] = float (level[2]), ia[1] = amplitude[2];
222  il[2] = float (level[6]), ia[2] = amplitude[6];
223  il[3] = float (level[1]), ia[3] = amplitude[1];
224  ik_x = float (i - SHRED_WIDTH / 2) * 0.0625f;
225  ik_y = float (j + SHRED_WIDTH / 2) * 0.0625f;
226  } else if (j >= SHRED_WIDTH / 2 && i < SHRED_WIDTH / 2) {
227  il[0] = float (level[4]), ia[0] = amplitude[4];
228  il[1] = float (level[0]), ia[1] = amplitude[0];
229  il[2] = float (level[8]), ia[2] = amplitude[8];
230  il[3] = float (level[3]), ia[3] = amplitude[3];
231  ik_x = float (i + SHRED_WIDTH / 2) * 0.0625f;
232  ik_y = float (j - SHRED_WIDTH / 2) * 0.0625f;
233  } else {
234  il[0] = float (level[0]), ia[0] = amplitude[0];
235  il[1] = float (level[5]), ia[1] = amplitude[5];
236  il[2] = float (level[2]), ia[2] = amplitude[2];
237  il[3] = float (level[8]), ia[3] = amplitude[8];
238  ik_x = float (i - SHRED_WIDTH / 2) * 0.0625f;
239  ik_y = float (j - SHRED_WIDTH / 2) * 0.0625f;
240  }
241  il[0] = il[0] * ik_y + (1.0f - ik_y) * il[3];
242  il[1] = il[1] * ik_y + (1.0f - ik_y) * il[2];
243  const float interp_level = il[0]*(1.0f - ik_x) + il[1] * ik_x;
244 
245  ia[0] = ia[0] * ik_y + (1.0f - ik_y) * ia[3];
246  ia[1] = ia[1] * ik_y + (1.0f - ik_y) * ia[2];
247  const float interp_amplitude = ia[0]*(1.0f-ik_x) + ia[1]*ik_x;
248 
249  const ushort h = qBound(2,
250  short (interp_level + FinalNoise(
251  latitude *16 + i,
252  longitude*16 + j) * interp_amplitude) - depth,
253  HEIGHT - 2);
254  short dirt_h;
255  if (h < 80) {
256  dirt_h = 5;
257  dirt_h+= short( 3.0f * FinalNoise(
258  ( latitude *16+i )*4,
259  ( longitude*16+j )*4 ) );
260  } else if (h < 100) {
261  dirt_h = (100 - h) * 6 / 20;
262  dirt_h+= short( 3.0f * FinalNoise(
263  ( latitude *16+i )*4,
264  ( longitude*16+j )*4 ) );
265  if( dirt_h < 0 ) {
266  dirt_h = 0;
267  }
268  } else {
269  dirt_h = 0;
270  }
271 
272  ushort k;
273  for (k = 1; k < h - dirt_h && h < HEIGHT - 1; ++k) {
274  PutNormalBlock(STONE, i, j, k);
275  }
276  for (; k < h; ++k) {
277  PutNormalBlock(sub, i, j, k);
278  }
279  }
