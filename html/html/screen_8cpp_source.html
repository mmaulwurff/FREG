241 void Screen::ControlPlayer(const int ch) {
242  CleanFileToShow();
243  if ( 'Q'==ch ) {
244  emit ExitReceived();
245  return;
246  }
247  if ( ch>='a' && ch<='z' ) { //actions with inventory
248  InventoryAction(ch-'a');
249  return;
250  }
251  switch ( ch ) { //interactions with world
252  case KEY_UP:
253  if ( player->GetDir()==NORTH ) {
254  player->Move(NORTH);
255  } else {
256  player->SetDir(NORTH);
257  }
258  break;
259  case KEY_DOWN:
260  if ( player->GetDir()==SOUTH ) {
261  player->Move(SOUTH);
262  } else {
263  player->SetDir(SOUTH);
264  }
265  break;
266  case KEY_RIGHT:
267  if ( player->GetDir()==EAST ) {
268  player->Move(EAST);
269  } else {
270  player->SetDir(EAST);
271  }
272  break;
273  case KEY_LEFT:
274  if ( player->GetDir()==WEST ) {
275  player->Move(WEST);
276  } else {
277  player->SetDir(WEST);
278  }
279  break;
280  case KEY_END: player->Move(DOWN); break;
281  case ' ': player->Jump(); break;
282  case '=': player->Move(); break;
283 
284  case '>':
285  player->SetDir(World::TurnRight(player->GetDir()));
286  break;
287  case '<':
288  player->SetDir(World::TurnLeft(player->GetDir()));
289  break;
290  case KEY_NPAGE: player->SetDir(DOWN); break;
291  case KEY_PPAGE: player->SetDir(UP); break;
292 
293  case KEY_HOME: player->Backpack(); break;
294  case 8:
295  case KEY_BACKSPACE: { //damage
296  ushort x, y, z;
297  ActionXyz(x, y, z);
298  player->Damage(x, y, z);
299  } break;
300  case 13:
301  case '\n': { //use
302  ushort x, y, z;
303  ActionXyz(x, y, z);
304  player->Use(x, y, z);
305  } break;
306  case '?': { //examine
307  ushort x, y, z;
308  ActionXyz(x, y, z);
309  player->Examine(x, y, z);
310  } break;
311  case '~': { //inscribe
312  ushort x, y, z;
313  ActionXyz(x, y, z);
314  player->Inscribe(x, y, z);
315  } break;
316  case 27: /*esc*/ player->StopUseAll(); break;
317 
318  case 'U': SetActionMode(USE); break;
319  case 'T': SetActionMode(THROW); break;
320  case 'O': SetActionMode(OBTAIN); break;
321  case 'W': SetActionMode(WIELD); break;
322  case 'I': SetActionMode(INSCRIBE); break;
323  case 'E': SetActionMode(EAT); break;
324  case 'B': SetActionMode(BUILD); break;
325  case 'C': SetActionMode(CRAFT); break;
326  case 'F': SetActionMode(TAKEOFF); break;
327 
328  case KEY_HELP:
329  case 'H':
330  wstandend(rightWin);
331  PrintFile(rightWin, "help.txt");
332  break;
333 
334  case ';': {
335  Inventory * const inv=player->PlayerInventory();
336  if ( inv ) {
337  player->MoveInsideInventory(
338  inv->Start(), inv->Size()-1);
339  }
340  } break;
341  case '-': shiftFocus = -!shiftFocus; break; //move focus down
342  case '+': shiftFocus = !shiftFocus; break; //move focus up
343 
344  case '!':
345  player->SetCreativeMode( !player->GetCreativeMode() );
346  break;
347  case ':': //command mode
348  PassString(command);
349  //no break
350  case '.':
351  player->ProcessCommand(command);
352  break;
353 
354  case 'L': RePrint(); break;
355 
356  case KEY_MOUSE: MouseAction(); break;
357  case 'R':
358  if ( player->GetCreativeMode() ) {
359  break;
360  }
361  player->SetActiveHand(!player->IsRightActiveHand());
362  Notify(tr("Now %1 hand is active.").
363  arg(tr(player->IsRightActiveHand() ?
364  "right" : "left")));
365  break;
366  default:
367  Notify(tr(
368  "Don't know what such key means: %1 ('%2').").
369  arg(ch).
370  arg(char(ch)));
371  }
372  updated=false;
373 } //Screen::ControlPlayer
