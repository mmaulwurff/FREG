140  void Workbench::Craft() {
141  for (int i=0; i<Start(); ++i) { // remove previous products
142  while ( Number(i) ) {
143  Block * const to_pull = ShowBlock(i);
144  Pull(i);
145  block_manager.DeleteBlock(to_pull);
146  }
147  }
148  int materials_number = 0;
149  for (int i=Start(); i<Size(); ++i) {
150  if ( Number(i) ) {
151  ++materials_number;
152  }
153  }
154  CraftList list(materials_number);
155  for (int i=Start(); i<Size(); ++i) {
156  if ( Number(i) ) {
157  list << new CraftItem(
158  {Number(i), ShowBlock(i)->Kind(), ShowBlock(i)->Sub()} );
159  }
160  }
161  if ( craft_manager.Craft(&list, Sub()) ) {
162  for (int i=0; i<list.size(); ++i) {
163  for (int n=0; n<list.at(i)->num; ++n) {
164  const CraftItem * const item = list.at(i);
165  GetExact(BlockManager::NewBlock(item->kind, item->sub), i);
166  }
167  }
168  }

171  bool Workbench::Drop(const int src, const int dest,
172  const int num, Inventory * const inv_to)
173  {
174  if ( inv_to == nullptr
175  || src >= Size()
176  || dest >= inv_to->Size()
177  || Number(src) == 0 )
178  {
179  return false;
180  }
181  for (int i=0; i<num; ++i) {
182  if ( not inv_to->Get(ShowBlock(src), dest) ) return false;
183  Pull(src);
184  if ( src < Start() ) {
185  // remove materials:
186  for (int i=Start(); i<Size(); ++i) {
187  while ( Number(i) ) {
188  Block * const to_pull = ShowBlock(i);
189  Pull(i);
190  block_manager.DeleteBlock(to_pull);
191  }
192  }
193  } else {
194  Craft();
195  }
196  }
197  return true;
198  }

275  void Converter::DoRareAction() {
276  if ( isOn && fuelLevel < SECONDS_IN_DAY/DamageLevel() ) {
277  for (int i=Size()-1; i>=0; --i) {
278  Block * const block = ShowBlock(i);
279  if ( block != nullptr ) {
280  const int add = ConvertRatio(block->Sub());
281  if ( add > 0 ) {
282  fuelLevel += add;
283  Pull(i);
284  block_manager.DeleteBlock(block);
285  break;
286  }
287  }
288  }
289  }
290  World * const world = GetWorld();
291  if ( fuelLevel <= 0
292  || ( Sub() == STONE
293  && world->GetBlock(X(), Y(), Z()+1)->Sub() == WATER ) )
294  {
295  Damage(1, damageKindOff);
296  } else {
297  if (world->Damage(X(), Y(), Z()+1, DamageLevel(), damageKindOn)<=0)
298  {
299  world->DestroyAndReplace(X(), Y(), Z()+1);
300  }
301  fuelLevel -= DamageLevel();
302  }
303  }
