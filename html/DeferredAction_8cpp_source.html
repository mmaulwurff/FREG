43 void DeferredAction::Build() const {
44  int z = z_self;
45  if ( DOWN==attachedBlock->GetDir() &&
46  AIR!=GetWorld()->GetBlock(x_self, y_self, z_self)->Sub() )
47  {
48  if ( world->Move(attachedBlock->X(), attachedBlock->Y(),
49  attachedBlock->Z(), UP) )
50  {
51  ++z;
52  } else {
53  return;
54  }
55  }
56  const int id = material->GetId();
57  if ( not world->Build(material, x_self, y_self, z,
58  World::TurnRight(attachedBlock->GetDir()), attachedBlock) )
59  { // build not successful
60  return;
61  } // else:
62  Inventory * const inv = attachedBlock->HasInventory();
63  if ( inv == nullptr ) {
64  return;
65  } // else:
66  inv->Pull(srcSlot);
67  // put more material in building inventory slot:
68  if ( inv->Number(srcSlot) ) {
69  return;
70  } // else:
71  for (int i=srcSlot+1; i<inv->Size() &&
72  inv->Number(srcSlot)<MAX_STACK_SIZE; ++i)
73  {
74  const Block * const block_i = inv->ShowBlock(i);
75  if ( block_i && id==block_i->GetId() ) {
76  inv->MoveInside(i, srcSlot, inv->Number(i));
77  }
78  }
79 } // void DeferredAction::Build()
