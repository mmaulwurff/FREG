42 void DeferredAction::Build() {
43  if ( DOWN==attachedBlock->GetDir() &&
44  AIR!=world->Sub(xTarg, yTarg, zTarg) )
45  {
46  if ( world->Move(
47  attachedBlock->X(),
48  attachedBlock->Y(),
49  attachedBlock->Z(),
50  UP) )
51  {
52  ++zTarg;
53  } else {
54  return;
55  }
56  }
57  const int kind=material->Kind();
58  const int sub =material->Sub();
59  if ( !world->Build(material,
60  xTarg, yTarg, zTarg,
61  World::TurnRight(attachedBlock->GetDir()),
62  attachedBlock) )
63  { // build not successful
64  return;
65  }
66  Inventory * const inv=attachedBlock->HasInventory();
67  if ( !inv ) {
68  return;
69  }
70  inv->Pull(srcSlot);
71  //put more material in building
72  //inventory slot:
73  if ( inv->Number(srcSlot) ) {
74  return;
75  }
76  for (ushort i=srcSlot+1; i<inv->Size() &&
77  inv->Number(srcSlot)<MAX_STACK_SIZE; ++i)
78  {
79  const Block * const block_i=inv->ShowBlock(i);
80  if ( block_i &&
81  kind==block_i->Kind() &&
82  sub ==block_i->Sub() )
83  {
84  inv->MoveInside(i, srcSlot, inv->Number(i));
85  }
86  }
87 }
