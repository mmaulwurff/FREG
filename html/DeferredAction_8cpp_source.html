46 void DeferredAction::Build() const {
47  int z = Z();
48  World * const world = attachedBlock->GetWorld();
49  if ( DOWN==attachedBlock->GetDir() &&
50  AIR!=world->GetBlock(X(), Y(), Z())->Sub() )
51  {
52  if ( world->Move(attachedBlock->X(),
53  attachedBlock->Y(), attachedBlock->Z(), UP) )
54  {
55  ++z;
56  } else {
57  return;
58  }
59  }
60  const int id = material->GetId();
61  if ( not world->Build(material, X(), Y(), z,
62  World::TurnRight(attachedBlock->GetDir()), attachedBlock) )
63  { // build not successful
64  return;
65  } // else:
66  Inventory * const inv = attachedBlock->HasInventory();
67  if ( inv == nullptr ) return;
68  inv->Pull(srcSlot);
69  // put more material in building inventory slot:
70  if ( inv->Number(srcSlot) ) return;
71  for (int i=srcSlot+1; i<inv->Size() &&
72  inv->Number(srcSlot)<MAX_STACK_SIZE; ++i)
73  {
74  const Block * const block_i = inv->ShowBlock(i);
75  if ( block_i && id==block_i->GetId() ) {
76  inv->MoveInside(i, srcSlot, inv->Number(i));
77  }
78  }
79 } // void DeferredAction::Build()
