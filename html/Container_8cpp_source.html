111 // Workbench::
112  void Workbench::Craft() {
113  for (int i=0; i<Start(); ++i) { // remove previous products
114  while ( Number(i) ) {
115  Block * const to_pull = ShowBlock(i);
116  Pull(i);
117  block_manager.DeleteBlock(to_pull);
118  }
119  }
120  int materials_number = 0;
121  for (int i=Start(); i<Size(); ++i) {
122  if ( Number(i) ) {
123  ++materials_number;
124  }
125  }
126  CraftList list(materials_number, 0);
127  for (int i=Start(); i<Size(); ++i) {
128  if ( Number(i) ) {
129  list << new CraftItem({Number(i), ShowBlock(i)->GetId()});
130  }
131  }
132  CraftList * products = world->GetCraftManager()->Craft(&list, Sub());
133  if ( products != nullptr ) {
134  for (int i=0; i<products->GetSize(); ++i) {
135  for (int n=0; n<products->GetItem(i)->num; ++n) {
136  int id = products->GetItem(i)->id;
137  GetExact(block_manager.NewBlock(
138  block_manager.KindFromId(id),
139  block_manager. SubFromId(id)), i);
140  }
141  }
142  delete products;
143  }
144  }

146  bool Workbench::Drop(const int src, const int dest,
147  const int num, Inventory * const inv_to)
148  {
149  if ( inv_to == nullptr
150  || src >= Size()
151  || dest >= inv_to->Size()
152  || Number(src) == 0 )
153  {
154  return false;
155  }
156  for (int i=0; i<num; ++i) {
157  if ( not inv_to->Get(ShowBlock(src), dest) ) return false;
158  Pull(src);
159  if ( src < Start() ) {
160  // remove materials:
161  for (int i=Start(); i<Size(); ++i) {
162  while ( Number(i) ) {
163  Block * const to_pull = ShowBlock(i);
164  Pull(i);
165  block_manager.DeleteBlock(to_pull);
166  }
167  }
168  } else {
169  Craft();
170  }
171  }
172  return true;
173  }
