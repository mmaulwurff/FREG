43 bool Shred::LoadShred() {
44  QByteArray * const data = GetWorld()->GetShredData(longitude, latitude);
45  if ( data == nullptr ) return false;
46  QDataStream in(*data);
47  quint8 dataStreamVersion;
48  quint8 shredFormatVersion;
49  in >> dataStreamVersion >> shredFormatVersion;
50  if ( Q_UNLIKELY(DATASTREAM_VERSION != dataStreamVersion) ) {
51  fprintf(stderr,
52  "QDataStream version: %d (must be %d). Generating new shred.\n",
53  dataStreamVersion, DATASTREAM_VERSION);
54  return false;
55  } // else:
56  if ( Q_UNLIKELY(CURRENT_SHRED_FORMAT_VERSION != shredFormatVersion) ) {
57  fprintf(stderr,
58  "Shred format version: %d (must be %d).\nGenerating new shred.\n",
59  shredFormatVersion, CURRENT_SHRED_FORMAT_VERSION);
60  return false;
61  } // else:
62  in.setVersion(DATASTREAM_VERSION);
63  Block * const null_stone = Normal(NULLSTONE);
64  Block * const air = Normal(AIR);
65  memset(lightMap, 0,
66  sizeof(lightMap[0][0][0]) * SHRED_WIDTH * SHRED_WIDTH * HEIGHT);
67  for (ushort x=0; x<SHRED_WIDTH; ++x)
68  for (ushort y=0; y<SHRED_WIDTH; ++y) {
69  PutBlock(null_stone, x, y, 0);
70  for (ushort z=1; ; ++z) {
71  quint8 kind, sub;
72  const bool normal = block_manager.KindSubFromFile(in, kind, sub);
73  if ( sub==SKY || sub==STAR ) {
74  for ( ; z < HEIGHT-1; ++z) {
75  PutBlock(air, x, y, z);
76  }
77  PutBlock(Normal(sub), x, y, HEIGHT-1);
78  lightMap[x][y][HEIGHT-1] = 1;
79  break;
80  } else if ( normal ) {
81  PutBlock(Normal(sub), x, y, z);
82  } else {
83  SetBlockNoCheck(block_manager.BlockFromFile(
84  in, kind, sub), x, y, z);
85  }
86  }
87  }
88  return true;
89 } // bool Shred::LoadShred()

160 void Shred::PhysEventsFrequent() {
161  // falling
162  for (auto i = fallList.begin(); i != fallList.end(); ) {
163  if ( (*i)->GetShred() != this ) {
164  i = fallList.erase(i);
165  continue;
166  } // else:
167  const ushort weight = (*i)->Weight();
168  if ( weight ) {
169  // PhysEventsFrequent is the first function in physics turn,
170  // so *i always belongs to this shred,
171  // so CoordInShred will return correct coordinates in this shred.
172  const ushort x = (*i)->X();
173  const ushort y = (*i)->Y();
174  const ushort z = (*i)->Z();
175  if ( LIQUID == GetBlock(CoordInShred(x),
176  CoordInShred(y), z-1)->Kind() )
177  {
178  (*i)->SetFalling(false);
179  i = fallList.erase(i);
180  } else if ( weight <= GetBlock(CoordInShred(x),
181  CoordInShred(y), z-1)->Weight()
182  || !GetWorld()->Move(x, y, z, DOWN) )
183  {
184  (*i)->FallDamage();
185  (*i)->SetFalling(false);
186  const short durability = (*i)->GetDurability();
187  i = fallList.erase(i);
188  if ( durability <= 0 ) {
189  GetWorld()->DestroyAndReplace(x, y, z);
190  }
191  if ( GetBlock(CoordInShred(x),
192  CoordInShred(y), z-1)->GetDurability() <= 0 )
193  {
194  GetWorld()->DestroyAndReplace(x, y, z-1);
195  }
196  } else {
197  ++i;
198  }
199  } else {
200  (*i)->SetFalling(false);
201  i = fallList.erase(i);
202  }
203  }
204  // frequent actions
205  for (auto j = activeListFrequent.constBegin();
206  j != activeListFrequent.constEnd(); ++j)
207  {
208  if ( (*j)->GetShred() == this ) {
209  (*j)->ActFrequent();
210  }
211  }
212  UnregisterExternalActives();
213 } // void Shred::PhysEventsFrequent()
