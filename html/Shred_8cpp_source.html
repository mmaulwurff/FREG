36 bool Shred::LoadShred() {
37  const QByteArray * const data =
38  GetWorld()->GetShredData(longitude, latitude);
39  if ( data == nullptr ) return false;
40  QDataStream in(*data);
41  quint8 dataStreamVersion;
42  quint8 shredFormatVersion;
43  in >> dataStreamVersion >> shredFormatVersion;
44  if ( Q_UNLIKELY(DATASTREAM_VERSION != dataStreamVersion) ) {
45  fprintf(stderr, "%s: %d (must be %d). Generating new shred.\n",
46  Q_FUNC_INFO, dataStreamVersion, DATASTREAM_VERSION);
47  return false;
48  } // else:
49  if ( Q_UNLIKELY(CURRENT_SHRED_FORMAT_VERSION != shredFormatVersion) ) {
50  fprintf(stderr,
51  "%s: Shred format: %d (must be %d). Generating new shred.\n",
52  Q_FUNC_INFO, shredFormatVersion, CURRENT_SHRED_FORMAT_VERSION);
53  return false;
54  } // else:
55  in.setVersion(DATASTREAM_VERSION);
56  in >> (quint8 &)(type);
57  for (int i=0; i<=TIME_EVENING; ++i) {
58  quint8 new_weather;
59  in >> new_weather;
60  weather[i] = static_cast<weathers>(new_weather);
61  }
62  Block * const null_stone = Normal(NULLSTONE);
63  Block * const air = Normal(AIR);
64  SetAllLightMapNull();
65  for (int x=0; x<SHRED_WIDTH; ++x)
66  for (int y=0; y<SHRED_WIDTH; ++y) {
67  PutBlock(null_stone, x, y, 0);
68  for (int z=1; ; ++z) {
69  int kind, sub;
70  const bool normal = block_manager.KindSubFromFile(in, &kind, &sub);
71  if ( normal ) {
72  if ( sub==SKY || sub==STAR ) {
73  for ( ; z < HEIGHT-1; ++z) {
74  PutBlock(air, x, y, z);
75  }
76  PutBlock(Normal(sub), x, y, HEIGHT-1);
77  break;
78  } else {
79  PutBlock(Normal(sub), x, y, z);
80  }
81  } else {
82  SetBlockNoCheck(block_manager.BlockFromFile(in, kind, sub),
83  x, y, z);
84  }
85  }
86  }
87  return true;
88 } // bool Shred::LoadShred()

193  world->DestroyAndReplace(x, y, z-1);
194  }
195  *i = nullptr;
196  }
197  }
198  }
199  for (auto i = activeListFrequent.constBegin();
200  i != activeListFrequent.constEnd(); ++i)

202  if ( *i != nullptr ) {
203  (*i)->ActFrequent();
204  }

637  }

639  SetBlock(Normal(WOOD), x+1, y+1, z-1);
640  }
641  for (int k=z; k < z+height-1; ++k) { // trunk
642  PutBlock(Normal(WOOD), x+1, y+1, k);
643  }
644  // branches
645  const int r = qrand();
