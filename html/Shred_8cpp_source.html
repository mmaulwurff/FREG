41 bool Shred::LoadShred() {
42  const QByteArray * const data =
43  GetWorld()->GetShredData(longitude, latitude);
44  if ( data == nullptr ) return false;
45  QDataStream in(*data);
46  quint8 shredFormatVersion;
47  in >> shredFormatVersion;
48  if ( Q_UNLIKELY(CURRENT_SHRED_FORMAT_VERSION != shredFormatVersion) ) {
49  fprintf(stderr,
50  "%s: Shred format: %d (must be %d). Generating new shred.\n",
51  Q_FUNC_INFO, shredFormatVersion, CURRENT_SHRED_FORMAT_VERSION);
52  return false;
53  } // else:
54  in.setVersion(DATASTREAM_VERSION);
55  quint8 read_type;
56  in >> read_type;
57  type = static_cast<shred_type>(read_type);
58  Block * const null_stone = Normal(NULLSTONE);
59  Block * const air = Normal(AIR);
60  SetAllLightMapNull();
61  for (int x=0; x<SHRED_WIDTH; ++x)
62  for (int y=0; y<SHRED_WIDTH; ++y) {
63  PutBlock(null_stone, x, y, 0);
64  for (int z=1; ; ++z) {
65  quint8 kind, sub;
66  if ( BlockManager::KindSubFromFile(in, &kind, &sub) ) { // normal
67  if ( sub==SKY || sub==STAR ) {
68  while (z < HEIGHT-1) {
69  PutBlock(air, x, y, z++);
70  }
71  PutBlock(Normal(sub), x, y, HEIGHT-1);
72  break;
73  } else {
74  PutBlock(Normal(sub), x, y, z);
75  }
76  } else {
77  Active * const active = (blocks[x][y][z] =
78  BlockManager::BlockFromFile(in, kind, sub))->ActiveBlock();
79  if ( active != nullptr ) {
80  active->SetXyz(
81  (ShredX() << SHRED_WIDTH_SHIFT) | x,
82  (ShredY() << SHRED_WIDTH_SHIFT) | y, z );
83  RegisterInit(active);
84  Falling * const falling = active->ShouldFall();
85  if ( falling != nullptr && falling->IsFalling() ) {
86  fallList.append(falling);
87  }
88  }
89  }
90  }
91  }
92  return true;
93 } // bool Shred::LoadShred()

150 Shred::~Shred() {
151  QByteArray * const shred_data = new QByteArray();
152  shred_data->reserve(30000);
153  QDataStream outstr(shred_data, QIODevice::WriteOnly);
154  outstr << CURRENT_SHRED_FORMAT_VERSION;
155  outstr.setVersion(DATASTREAM_VERSION);
156  outstr << quint8(GetTypeOfShred());
157  for (int x=0; x<SHRED_WIDTH; ++x)
158  for (int y=0; y<SHRED_WIDTH; ++y) {
159  int height = HEIGHT-2;
160  for ( ; blocks[x][y][height]->Sub()==AIR; --height);
161  for (int z=1; z <= height; ++z) {
162  Block * const block = blocks[x][y][z];
163  if ( block == Normal(block->Sub()) ) {
164  block->SaveNormalToFile(outstr);
165  } else {
166  block->SaveToFile(outstr);
167  delete block;
168  }
169  }
170  blocks[x][y][HEIGHT-1]->SaveNormalToFile(outstr);
171  }
172  GetWorld()->SetShredData(shred_data, longitude, latitude);
173 }

577  for (int k=leaves_level; k<z+height; ++k ) {
578  PutBlock(leaves, i, j, k);
579  }
580  }
581  for (int k=qMax(z-1, 1); k < z+height-1; ++k) { // trunk
582  SetBlock(Normal(WOOD), x+1, y+1, k);
583  }
584  // branches
585  const int r = qrand();
586  if ( r & 0x1 ) SetNewBlock(BLOCK, WOOD, x, y+1, leaves_level, WEST);
587  if ( r & 0x2 ) SetNewBlock(BLOCK, WOOD, x+2, y+1, leaves_level, EAST);
588  if ( r & 0x4 ) SetNewBlock(BLOCK, WOOD, x+1, y, leaves_level, NORTH);
589  if ( r & 0x8 ) SetNewBlock(BLOCK, WOOD, x+1, y+2, leaves_level, SOUTH);
590  return true;
