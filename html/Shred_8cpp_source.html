35 bool Shred::LoadShred(QFile & file) {
36  QDataStream in(qUncompress(file.readAll()));
37  quint8 version;
38  in >> version;
39  if ( Q_UNLIKELY(DATASTREAM_VERSION!=version) ) {
40  fprintf(stderr,
41  "Wrong version: %d\nGenerating new shred.\n",
42  DATASTREAM_VERSION);
43  return false;
44  }
45  in.setVersion(DATASTREAM_VERSION);
46  for (ushort i=0; i<SHRED_WIDTH; ++i)
47  for (ushort j=0; j<SHRED_WIDTH; ++j) {
48  PutNormalBlock(NULLSTONE, i, j, 0);
49  lightMap[i][j][0]=0;
50  for (ushort k=1; k<HEIGHT-1; ++k) {
51  SetBlock(block_manager.BlockFromFile(in), i, j, k);
52  lightMap[i][j][k]=0;
53  }
54  SetBlock(block_manager.BlockFromFile(in), i, j, HEIGHT-1);
55  lightMap[i][j][HEIGHT-1]=1;
56  }
57  return true;
58 }

467 bool Shred::Tree(const ushort x, const ushort y, const ushort z,
468  const ushort height)
469 {
470  if (
471  SHRED_WIDTH<=x+2 ||
472  SHRED_WIDTH<=y+2 ||
473  HEIGHT-1<=z+height ||
474  height<2 )
475  {
476  return false;
477  }
478  ushort i, j, k;
479  for (i=x; i<=x+2; ++i)
480  for (j=y; j<=y+2; ++j)
481  for (k=z; k<z+height; ++k) {
482  if ( AIR!=Sub(i, j, k) && WATER!=Sub(i, j, k) ) {
483  return false;
484  }
485  }
486  for (k=z; k < z + height - 1; ++k) { //trunk
487  PutNormalBlock(WOOD, x+1, y+1, k);
488  }
489  if ( ENVIRONMENT==GetBlock(x+1, y+1, z-1)->Movable() ) {
490  block_manager.DeleteBlock(blocks[x+1][y+1][z-1]);
491  PutNormalBlock(WOOD, x+1, y+1, z-1);
492  }
493  //branches
494  if ( qrand()%2 ) SetNewBlock(BLOCK, WOOD, x, y+1, z+height/2, WEST);
495  if ( qrand()%2 ) SetNewBlock(BLOCK, WOOD, x+2, y+1, z+height/2, EAST);
496  if ( qrand()%2 ) SetNewBlock(BLOCK, WOOD, x+1, y, z+height/2, NORTH);
497  if ( qrand()%2 ) SetNewBlock(BLOCK, WOOD, x+1, y+2, z+height/2, SOUTH);
498  //leaves
499  for (i=x; i<=x+2; ++i)
500  for (j=y; j<=y+2; ++j)
501  for (k=z+height/2; k<z+height; ++k) {
502  if ( AIR==Sub(i, j, k) ) {
503  PutNormalBlock(GREENERY, i, j, k);
504  }
505  }
506  return true;
507 } //Shred::Tree
