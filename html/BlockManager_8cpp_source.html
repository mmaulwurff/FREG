74 Block * BlockManager::BlockFromFile(QDataStream & str) {
75  quint8 data;
76  str >> data;
77  const quint8 sub=(data & 0x7F);
78  if ( data & 0x80 ) {
79  return NormalBlock(sub);
80  }
81  quint8 kind;
82  str >> kind;
83  //if some kind will not be listed here,
84  //blocks of this kind just will not load,
85  //unless kind is inherited from Inventory class or one
86  //of its derivatives - in this case this may cause something bad.
87  switch ( kind ) {
88  case BLOCK: return New<Block >(str, sub);
89  case PICK: return New<Pick >(str, sub);
90  case PLATE: return New<Plate >(str, sub);
91  case LADDER: return New<Ladder>(str, sub);
92  case WEAPON: return New<Weapon>(str, sub);
93  case BUSH: return New<Bush >(str, sub);
94  case CHEST: return New<Chest >(str, sub);
95  case RABBIT: return New<Rabbit>(str, sub);
96  case DWARF: return New<Dwarf >(str, sub);
97  case PILE: return New<Pile >(str, sub);
98  case GRASS: return New<Grass >(str, sub);
99  case ACTIVE: return New<Active>(str, sub);
100  case LIQUID: return New<Liquid>(str, sub);
101  case LOCKED_DOOR:
102  case DOOR: return New<Door >(str, sub);
103  case CLOCK: return New<Clock >(str, sub);
104  case CREATOR: return New<Creator>(str, sub);
105  case WORKBENCH: return New<Workbench>(str, sub);
106  default:
107  fprintf(stderr,

109  kind);
