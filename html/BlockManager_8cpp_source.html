205 bool BlockManager::KindSubFromFile(QDataStream & str,
206  quint8 * kind, quint8 * sub)
207 {
208  str >> *sub;
209  if ( *sub & 0x80 ) { // normal bit
210  *sub &= 0x7F;
211  return true;
212  } else {
213  *sub &= 0x7F;
214  str >> *kind;
215  return false;
216  }
