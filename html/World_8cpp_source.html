241  emit UpdatedAll();

244  return;
245  // while reloading, reuse memory places.
246  case NORTH:
247  emit StartMove(toResetDir);

311 void World::PhysEvents() {
312  Lock();
313  static const int start = NumShreds()/2 - numActiveShreds/2;
314  static const int end = start + numActiveShreds;
315  for (int i=start; i<end; ++i)
316  for (int j=start; j<end; ++j) {
317  shreds[ShredPos(i, j)]->PhysEventsFrequent();
318  }
319 
320  if ( TimeStepsInSec() > timeStep ) {
321  ++timeStep;
322  } else {
323  for (int i=start; i<end; ++i)
324  for (int j=start; j<end; ++j) {
325  shreds[ShredPos(i, j)]->PhysEventsRare();
326  }
327  timeStep = 0;
328  ++time;
329  switch ( TimeOfDay() ) {
330  default: break;

332  emit Notify(tr("It's morning now."));
333  ReEnlightenTime();

337  case END_OF_EVENING:

339  ReEnlightenTime();
340  break;
341  }
342  }

345  Unlock();

347  // emit ExitReceived(); // close all after 1 turn

350 bool World::DirectlyVisible(
351  int x_from, int y_from, int z_from,
352  int x, int y, int z)
353 const {
354  /// optimized DDA line with integers only.
355  int max = Abs(Abs(x-=x_from) > Abs(y-=y_from) ? x : y);
356  if ( Abs(z-=z_from) > max) {
357  max = Abs(z);
358  }
359  x_from *= max;
360  y_from *= max;
361  z_from *= max;
362  for (int i=max-1; i-- > 0; ) {
363  if ( not (GetBlock((x_from+=x)/max, (y_from+=y)/max, (z_from+=z)/max)->
364  Transparent()
365  && GetBlock( (x_from+max-1)/max, (y_from+max-1)/max,
366  (z_from+max-1)/max )->Transparent()) )
367  {
368  return false;
369  }
370  }
371  return true;
372 }

407 bool World::CanMove(const int x, const int y, const int z,
408  const int newx, const int newy, const int newz, const dirs dir)
409 {
410  Block * const block = GetBlock(x, y, z);
411  Block * block_to = GetBlock(newx, newy, newz);
412  Falling * const falling = block->ShouldFall();
413  if ( DOWN != dir
414  && block->Weight() != 0
415  && falling != nullptr
416  && falling->IsFalling()
417  && AIR == GetBlock(x, y, z-1)->Sub()
418  && AIR == GetBlock(newx, newy, newz-1)->Sub() )
419  { // prevent moving while falling
420  return false;
421  }
422  const push_reaction reaction = block->PushResult(dir);
423  switch ( reaction ) {
424  case MOVABLE:
425  if ( Damage(newx, newy, newz, 1, DAMAGE_PUSH_UP + dir) <= 0 ) {
426  DestroyAndReplace (newx, newy, newz);
427  block_to = GetBlock(newx, newy, newz);
428  }
429  break;
430  case ENVIRONMENT:
431  if ( *block != *block_to ) { // prevent useless flow
432  break;
433  } // no break;
434  default: return false;
435  }
436  switch ( block_to->PushResult(dir) ) {
437  case MOVABLE:
438  return ( block->Weight() > block_to->Weight()
439  && Move(newx, newy, newz, dir) );
440  case ENVIRONMENT: return true;
441  case NOT_MOVABLE: break;
442  case MOVE_UP:
443  if ( dir > DOWN ) { // not UP and not DOWN
444  Move(x, y, z, UP);
445  }
446  break;
447  case JUMP:
448  if ( dir > DOWN ) { // not UP and not DOWN
449  Jump(x, y, z, dir);
450  }
451  break;
452  case DAMAGE:
453  if ( Damage(x, y, z, block_to->DamageLevel(), block_to->DamageKind())
454  <= 0 )
455  {
456  DestroyAndReplace(x, y, z);
457  }
458  break;
459  }
460  return false;
461 } // bool World::CanMove(const int x, y, z, newx, newy, newz, int dir)
