51  usage_types Weapon::Use(Block * const who) {
52  Shred * const shred = GetShred();
53  if ( shred != nullptr ) { // not currently in inventory
54  Inventory * const inv = who->HasInventory();
55  if ( inv != nullptr && inv->Get(this) ) {
56  Unregister();
