78 float WorldMap::Deg(const int x, const int y, const int size) {
79  const float x_cent = x-size/2.f;
80  const float y_cent = y-size/2.f;
81  float fi;
82  if ( x_cent > 0 && y_cent >= 0 ) {
83  fi = atanf(y_cent/x_cent);
84  } else if ( x_cent>0 && y_cent<0 ) {
85  fi = atanf(y_cent/x_cent)+2*PI;
86  } else if ( x_cent<0 ) {
87  fi = atanf(y_cent/x_cent)+PI;
88  } else if ( qFuzzyCompare(x_cent, 0.f) && y_cent>0 ) {
89  fi = PI/2;
90  } else if ( qFuzzyCompare(x_cent, 0.f) && y_cent<0 ) {
91  fi = 3*PI/2;
92  } else {
93  fi = 0;
94  }
95  return 360*fi / 2 / PI;
96 }

102 void WorldMap::Circle(
103  const int min_rad,
104  const int max_rad,
105  const char ch,
106  const int size,
107  char * const map)
108 {
109  Q_ASSERT(min_rad < max_rad);
110  float maxs[360] = { float(qrand()%(max_rad - min_rad) + min_rad) };
111  for (int x=1; x<360; ++x) {
112  maxs[x] = ( x > 345 ) ? // connect beginning and end of circle
113  maxs[x-1] + (maxs[0] - maxs[345]) / 15 :
114  qBound(float(min_rad),
115  maxs[x-1]+(qrand()%400-200)/200.f, float(max_rad));
116  }
117  for (int y=0; y<size; ++y)
118  for (int x=0; x<size; ++x) {
119  if ( R(x, y, size) < maxs[Round(Deg(x, y, size))] ) {
120  map[x*size+y] = ch;
121  }
122  }
123 }

125 void WorldMap::GenerateMap(
126  const QString world_name,
127  int size,
128  const char outer,
129  const int seed)
130 {
131  qsrand(seed);
132  size = qMax(10, size);
133 
134  char * const map = new char[size*size];
135  memset(map, outer, size*size);
136 
137  const float min_rad = size/3.0f;
138  const float max_rad = size/2.0f;
139  Circle(min_rad, max_rad, SHRED_WASTE, size, map);
140  Circle(min_rad/2, max_rad/2, SHRED_DEAD_FOREST, size, map);
141  Circle(min_rad/3, max_rad/3+1, SHRED_DEAD_HILL, size, map);
142  Circle(min_rad/4, max_rad/4+1, SHRED_MOUNTAIN, size, map);
143 
144  int lakes_number = (qrand() % size) + 5;
145  while ( lakes_number-- ) {
146  char type = SHRED_WATER;
147  switch ( qrand()%4 ) {
148  case 1: type = SHRED_ACID_LAKE; break;
149  case 2: type = SHRED_LAVA_LAKE; break;
150  case 3: type = SHRED_CRATER; break;
151  }
152  const float lake_size = qrand() % (size/10) + 1;
153  const int lake_start_x = qrand() % int(size-lake_size);
154  const int lake_start_y = qrand() % int(size-lake_size);
155  const int border = (lake_size-1)*(lake_size-1)/2/2;
156  for (int x=lake_start_x; x<lake_start_x+lake_size; ++x)
157  for (int y=lake_start_y; y<lake_start_y+lake_size; ++y) {
158  if ( (x-lake_start_x-lake_size/2)*(x-lake_start_x-lake_size/2)
159  + (y-lake_start_y-lake_size/2)*(y-lake_start_y-lake_size/2)
160  < border )
161  {
162  map[x*size+y] = type;
163  }
164  }
165  }
166 
167  long spawn_longitude, spawn_latitude;
168  MakeAndSaveSpawn(world_name, size, &spawn_longitude, &spawn_latitude);
169  PieceOfEden(spawn_latitude-1, spawn_longitude-1, map, size);
170 
171  FILE * const file = fopen(qPrintable(world_name+"/map.txt"), "wb");
172  for (int y=0; y<size; ++y, fputc('\n', file))
173  for (int x=0; x<size; ++x) {
174  fputc(map[x*size+y], file);
175  }
176  delete [] map;
177  fclose(file);
178 }
