56 float WorldMap::Deg(const int x, const int y, const int size) {
57  const float x_cent = x-size/2.f;
58  const float y_cent = y-size/2.f;
59  float fi;
60  if ( x_cent > 0 && y_cent >= 0 ) {
61  fi = atanf(y_cent/x_cent);
62  } else if ( x_cent>0 && y_cent<0 ) {
63  fi = atanf(y_cent/x_cent)+2*PI;
64  } else if ( x_cent<0 ) {
65  fi = atanf(y_cent/x_cent)+PI;
66  } else if ( qFuzzyCompare(x_cent, 0.f) && y_cent>0 ) {
67  fi = PI/2;
68  } else if ( qFuzzyCompare(x_cent, 0.f) && y_cent<0 ) {
69  fi = 3*PI/2;
70  } else {
71  fi = 0;
72  }
73  return 360*fi / 2 / PI;
74 }

80 void WorldMap::Circle(
81  const int min_rad,
82  const int max_rad,
83  const char ch,
84  const int size,
85  char * const map)
86 {
87  if ( min_rad >= max_rad ) {
88  fprintf(stderr, "%s: %c: min_rad (%d) >= max_rad (%d).\n",
89  Q_FUNC_INFO, ch, min_rad, max_rad);
90  }
91  float maxs[360] = { float(qMax(min_rad, qrand()%Round(max_rad))) };
92  for (int x=1; x<360; ++x) {
93  maxs[x] = qBound(float(min_rad),
94  maxs[x-1]+(qrand()%400-200)/200.f, float(max_rad));
95  if ( x > 315 ) { // connect beginning and end of circle
96  maxs[x] += (maxs[0]-maxs[x-1])/90;
97  }
98  }
99  for (int y=0; y<size; ++y)
100  for (int x=0; x<size; ++x) {
101  if ( R(x, y, size) < maxs[Round(Deg(x, y, size))] ) {
102  map[x*size+y] = ch;
103  }
104  }
105 }

124  Circle(min_rad/4, max_rad/4+1, SHRED_MOUNTAIN, size, map);
125

128  char type = SHRED_WATER;
129  switch ( qrand()%3 ) {
130  case 1: type = SHRED_ACID_LAKE; break;
131  case 2: type = SHRED_LAVA_LAKE; break;
132  }
133  const float lake_size = qrand() % (size/10) + 1;
134  const int lake_start_x = qrand() % int(size-lake_size);
135  const int lake_start_y = qrand() % int(size-lake_size);
136  const int border = (lake_size-1)*(lake_size-1)/2/2;
137  for (int x=lake_start_x; x<lake_start_x+lake_size; ++x)
138  for (int y=lake_start_y; y<lake_start_y+lake_size; ++y) {
139  if ( (x-lake_start_x-lake_size/2)*(x-lake_start_x-lake_size/2)
140  + (y-lake_start_y-lake_size/2)*(y-lake_start_y-lake_size/2)
141  < border )
142  {
143  map[x*size+y] = type;
144  }
145  }
146  }
147

149  for (int y=0; y<size; ++y, fputc('\n', file))
150  for (int x=0; x<size; ++x) {
151  fputc(map[x*size+y], file);
152  }
153  delete [] map;
154  fclose(file);
155 }
