75 void Dwarf::Damage(const int dmg, const int dmg_kind) {
76  if ( dmg_kind >= DAMAGE_PUSH_UP ) {
77  Active::Damage(dmg, DAMAGE_PUSH_UP);
78  return;
79  }
80  int damage_to_self = dmg;
81  const int places[] = { ON_HEAD, ON_BODY, ON_LEGS };
82  for (const int i : places) {
83  if ( Number(i) == 0 ) continue;
84  Block * const armour = ShowBlock(i);
85  const int dur_before_damage = armour->GetDurability();
86  const int damage_divider = (i == ON_BODY) ? 2 : 4;
87  armour->Damage(dmg/damage_divider, dmg_kind);
88  if ( armour->GetDurability() < dur_before_damage ) {
89  damage_to_self -= dmg/damage_divider;
90  if ( armour->GetDurability() <= 0 ) {
91  delete armour;
92  Pull(i);
93  }
94  }
95  }
96  Active::Damage(damage_to_self, dmg_kind);
97 }
