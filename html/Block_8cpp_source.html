67 void Block::Damage(const int dmg, const int dmg_kind) {
68  switch ( Sub() ) {
69  case GREENERY:
70  case HAZELNUT: durability = 0; return;
71  case NULLSTONE:
72  case STAR:
73  case AIR:
74  case SKY:
75  case SUN_MOON: return;
76  }
77  switch ( dmg_kind ) {
78  case NO_HARM: return;
79  case DAMAGE_ACID:
80  switch ( Sub() ) {
81  default: durability -= 2 * dmg; return;
82  case DIFFERENT: durability = 0; return;
83  case IRON:
84  case GLASS: return;
85  }
86  }
87  int mult = 1; // default
88  switch ( Sub() ) {
89  case DIFFERENT:
90  if ( TIME == dmg_kind ) {
91  durability = 0;
92  return;
93  }
94  return;
95  case WATER: mult = ( HEAT==dmg_kind || TIME==dmg_kind ); break;
96  case MOSS_STONE:
97  case STONE: switch ( dmg_kind ) {
98  case CRUSH:
99  case DAMAGE_HANDS:
100  case CUT: return;
101  case MINE: mult = 2; break;
102  } break;
103  case GLASS: durability = (HEAT==dmg_kind) ? durability : 0; return;
104  case WOOD: switch ( dmg_kind ) {
105  default: mult = 1; break;
106  case CUT: mult = 2; break;
107  case DAMAGE_HANDS: return;
108  } break;
109  case SAND:
110  case A_MEAT:
111  case H_MEAT: ++(mult = (THRUST==dmg_kind || HEAT==dmg_kind)); break;
112  case SOIL: switch ( dmg_kind ) {
113  case DIG: mult = 2; break;
114  case DAMAGE_FALL: return;
115  } break;
116  case FIRE: mult = (FREEZE==dmg_kind || TIME==dmg_kind); break;
117  }
118  durability -= mult*dmg;
119 }

224 void Block::SaveToFile(QDataStream & out) {
225  if ( this == block_manager.NormalBlock(sub) ) {
226  out << quint8( 0x80 | sub );
227  } else {
228  out << sub << quint8(BlockManager::KindFromId(id)) <<
229  (quint16)( ( ( ( durability
230  <<= 3 ) |= direction )
231  <<= 1 ) |= !!note );
232  if ( Q_UNLIKELY(note) ) {
233  out << *note;
234  }
235  SaveAttributes(out);
236  }
