75 void Block::Damage(const int dmg, int dmg_kind) {
76  if ( dmg_kind > DAMAGE_PUSH_UP ) {
77  dmg_kind = DAMAGE_PUSH_UP;
78  }
79  switch ( Sub() ) {
80  case SUB_DUST:
81  case GREENERY:
82  case ROSE:
83  case HAZELNUT: durability = 0; return;
84  case NULLSTONE:
85  case STAR:
86  case AIR:
87  case SKY:
88  case SUN_MOON: return;
89  }
90  switch ( dmg_kind ) {
91  case DAMAGE_NO: return;
92  case DAMAGE_TIME: durability -= dmg; return;
93  case DAMAGE_ACID:
94  switch ( Sub() ) {
95  default: durability -= 2 * dmg; return;
96  case DIFFERENT: durability = 0; return;
97  case ACID:
98  case ADAMANTINE:
99  case IRON:
100  case GLASS: return;
101  }
102  }
103  int mult = 1; // default
104  switch ( Sub() ) {
105  case DIFFERENT: return;
106  case MOSS_STONE:
107  case STONE: switch ( dmg_kind ) {
108  case DAMAGE_HEAT:
109  case DAMAGE_HANDS:
110  case DAMAGE_PUSH_UP:
111  case DAMAGE_CUT: return;
112  case DAMAGE_MINE: mult = 2; break;
113  } break;
114  case WOOD: switch ( dmg_kind ) {
115  case DAMAGE_CUT: mult = 2; break;
116  case DAMAGE_PUSH_UP:
117  case DAMAGE_HANDS: return;
118  } break;
119  case A_MEAT:
120  case H_MEAT:
121  mult += (DAMAGE_THRUST==dmg_kind || DAMAGE_HEAT==dmg_kind);
122  break;
123  case SAND:
124  case SOIL: mult += ( DAMAGE_DIG == dmg_kind ); break;
125  case ADAMANTINE:
126  case FIRE: mult = ( DAMAGE_FREEZE == dmg_kind ); break;
127  case WATER: mult = ( DAMAGE_HEAT == dmg_kind ); break;
128  case GLASS: switch ( dmg_kind ) {
129  case DAMAGE_PUSH_UP: durability -= dmg*((MAX_DURABILITY+9)/10); return;
130  default: durability = 0; // no break;
131  case DAMAGE_HEAT: return;
132  }
133  case IRON: switch ( dmg_kind ) {
134  case DAMAGE_HANDS:
135  case DAMAGE_PUSH_UP: return;
136  }
137  }
138  durability -= mult*dmg;
139 }

248  if ( this == block_manager.Normal(sub) ) {
249  SaveNormalToFile(out);
250  } else {
251  out << sub << kind <<
252  ( ( ( ( durability
