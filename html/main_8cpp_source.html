45 bool IsValidWorldName(const QString world_name) {
46  const QStringList invalid_world_names(QStringList()
47  << "blocks"
48  << "moc"
49  << "obj"
50  << "recipes"
51  << "texts");
52  if ( world_name.left(5)=="help_" ) {
53  printf("Invalid world name: %s.\n",
54  qPrintable(world_name));
55  return false;
56  }
57  for (auto i = invalid_world_names.constBegin();
58  i < invalid_world_names.constEnd(); ++i)
59  {
60  if ( *i == world_name ) {
61  printf("Invalid world name: %s.\n",
62  qPrintable(world_name));
63  return false;
64  }
65  }
66  return true;
67 }
