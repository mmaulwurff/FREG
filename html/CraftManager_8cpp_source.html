90 CraftManager::CraftManager() : size(0) {
91  QDir::current().mkdir("recipes");
92  const QStringList recipesNames = QDir("recipes").entryList();
93  if ( recipesNames.size() < 3 ) { // minimum: . and ..
94  recipesList = 0;
95  recipesSubsList = 0;
96  return;
97  } // else:
98  // array sizes are a bit more than needed, this is ok
99  recipesList = new QList<CraftList *>[recipesNames.size()];
100  recipesSubsList = new int[recipesNames.size()];
101  for (auto & recipeName : recipesNames) { // file level
102  const quint8 sub = BlockManager::StringToSub(recipeName);
103  if ( sub == LAST_SUB ) {
104  continue;
105  } // else:
106  recipesSubsList[size] = sub;
107  QFile file(QString("recipes/" + recipeName));
108  file.open(QIODevice::ReadOnly | QIODevice::Text);
109  int line = 0;
110  while ( not file.atEnd() ) { // line(recipe) level
111  QTextStream in(file.readLine(),
112  QIODevice::ReadOnly | QIODevice::Text);
113  ++line;
114  int materials_number, products_number;
115  in >> materials_number >> products_number;
116  if ( materials_number==0 || products_number==0 ) {
117  continue;
118  }
119  CraftList * const recipe =
120  new CraftList(materials_number, products_number);
121  bool ok = true;
122  while ( materials_number-- ) {
123  if ( not recipe->LoadItem(in) ) {
124  fprintf(stderr, "Recipe read error: %s::%d.\n",
125  qPrintable(recipeName), line);
126  ok = false;
127  break;
128  }
129  }
130  if ( not ok ) {
131  delete recipe;
132  continue;
133  } // else:
134  recipe->Sort();
135  while ( products_number-- ) {
136  if ( not recipe->LoadItem(in) ) {
137  fprintf(stderr, "Recipe read error: %s::%d.\n",
138  qPrintable(recipeName), line);
139  ok = false;
140  break;
141  }
142  }
143  if ( not ok ) {
144  delete recipe;
145  continue;
146  } // else:
147  recipesList[size].append(recipe);
148  }
149  ++size;
150  }
151 } // CraftManager::CraftManager()

183 CraftList * CraftManager::CraftSub(CraftList * const recipe, const int sub)
184 const {
185  recipe->Sort();
186  // find needed recipes list
187  int point = 0;
188  for ( ; point < size && recipesSubsList[point] != sub; ++point);
189  if ( point == size ) return nullptr; //recipes list for sub not found
190  // find recipe and copy products from it
191  for (int i=0; i<recipesList[point].size(); ++i) {
192  const CraftList & tried = *recipesList[point].at(i);
193  if ( tried == *recipe ) {
194  int size = tried.GetProductsNumber();
195  CraftList * result = new CraftList(size, 0);
196  for (; size; --size) {
197  *result << new CraftItem(*tried.GetItem(tried.GetSize()-size));
198  }
199  return result;
200  }
201  }
202  return nullptr; // suitable recipe not found
203 }
