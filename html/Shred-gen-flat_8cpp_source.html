76 void Shred::Water(const subs sub) {
77  subs shore;
78  switch ( sub ) {
79  case WATER: shore = SAND; break;
80  case AIR:
81  case STONE: shore = STONE; break;
82  default:
83  case ACID: shore = GLASS; break;
84  }
85  const int depth = CountShredTypeAround(GetTypeOfShred()) + 1;
86  NormalUnderground(depth, shore);
87  int z_start = HEIGHT/2-depth;
88  NormalCube(0,0,z_start++, SHRED_WIDTH,SHRED_WIDTH,1, shore); // bottom
89  const WorldMap * const map = GetWorld()->GetMap();
90  if ( type != map->TypeOfShred(longitude-1, latitude) ) { // north
91  NormalCube(0,0,z_start, SHRED_WIDTH,1,depth, shore);
92  }
93  if ( type != map->TypeOfShred(longitude+1, latitude) ) { // south
94  NormalCube(0,SHRED_WIDTH-1,z_start, SHRED_WIDTH,1,depth, shore);
95  }
96  if ( type != map->TypeOfShred(longitude, latitude+1) ) { // east
97  NormalCube(SHRED_WIDTH-1,0,z_start, 1,SHRED_WIDTH,depth, shore);
98  }
99  if ( type != map->TypeOfShred(longitude, latitude-1) ) { // west
100  NormalCube(0,0,z_start, 1,SHRED_WIDTH,depth, shore);
101  }
102  Block * const air = Normal(AIR);
103  for (int i=0; i<SHRED_WIDTH; ++i)
104  for (int j=0; j<SHRED_WIDTH; ++j)
105  for (int k=z_start; k<=HEIGHT/2; ++k) {
106  if ( shore != GetBlock(i, j, k)->Sub() ) {
107  if ( AIR == sub ) {
108  PutBlock(air, i, j, k);
109  } else {
110  SetNewBlock(LIQUID, sub, i, j, k);
111  }
112  }
113  }
114 }
