64 //make block emit shining
65 //receives only non-sun light, from 0 to F
66 void World::Shine(const ushort i, const ushort j, const ushort k,
67  const uchar level, const bool init)
68 {
69  if ( !InBounds(i, j, k) || 0==level ) {
70  return;
71  }
72  if ( INVISIBLE!=Transparent(i, j, k) ) {
73  if ( SetFireLightMap(level << 4, i, j, k) ) {
74  emit Updated(i, j, k);
75  }
76  if ( !init ) {
77  return;
78  }
79  }
80  Shine(i-1, j, k, level-1);
81  Shine(i+1, j, k, level-1);
82  Shine(i, j-1, k, level-1);
83  Shine(i, j+1, k, level-1);
84  Shine(i, j, k-1, level-1);
85  Shine(i, j, k+1, level-1);
86 }

88 void World::SunShineVertical(const short x, const short y, const short) {
89  /* 2 1 3
90  * * First, light goes down, then divides to 4 branches
91  * ^ | ^ to N-S-E-W, and goes up.
92  * | | |
93  * | | |
94  * |<v>|
95  * # */
96  ushort light_lev=MAX_LIGHT_RADIUS;
97  ushort z=HEIGHT-2;
98  for ( ; SetSunLightMap(light_lev, x, y, z); --z) {
99  emit Updated(x, y, z);
100  const ushort transparent=Transparent(x, y, z);
101  if ( BLOCK_TRANSPARENT==transparent ) {
102  --light_lev;
103  } else if ( BLOCK_OPAQUE==transparent ) {
104  break;
105  }
106  }
107  UpShine(x-1, y, z);
108  UpShine(x+1, y, z);
109  UpShine(x, y-1, z);
110  UpShine(x, y+1, z);
111 }
