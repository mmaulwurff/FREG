20 /**\file Lighting-inertia.cpp
21  * \brief This file provides simple (also mad) lighting for freg.
22  *
23  * It has light inertia, meaning that block, enlightened by outer
24  * source, will remain enlightened when light source is removed.
25  * Light is divided to sunlight and other light. Sunlight is
26  * changing over the day.
27  * LightMap is uchar:
28  * & 0xF0 bits are for non-sun light,
29  * & 0x0F bits for sun. */

62 /// Makes block emit shining.
63 /** Receives only non-sun light, from 0 to F. */
64 void World::Shine(const ushort i, const ushort j, const ushort k,
65  const uchar level, bool)
66 {
67  if ( not InBounds(i, j, k) || 0==level ) return;
68  if ( SetFireLightMap(level << 4, i, j, k) &&
69  INVISIBLE != GetBlock(i, j, k)->Transparent() )
70  {
71  emit Updated(i, j, k);
72  }
73  Shine(i-1, j, k, level-1);
74  Shine(i+1, j, k, level-1);
75  Shine(i, j-1, k, level-1);
76  Shine(i, j+1, k, level-1);
77  Shine(i, j, k-1, level-1);
78  Shine(i, j, k+1, level-1);
79 }

81 void World::SunShineVertical(const short x, const short y, short z,
82  uchar light_lev)
83 {
84  if ( GetEvernight() ) return;
85  /* 2 1 3
86  * * First, light goes down, then divides to 4 branches
87  * ^ | ^ to N-S-E-W, and goes up.
88  * | | |
89  * | | |
90  * |<v>|
91  * # */
92  const Shred * const shred = GetShred(x, y);
93  const ushort x_in = Shred::CoordInShred(x);
94  const ushort y_in = Shred::CoordInShred(y);
95  for ( ; SetSunLightMap(light_lev, x, y, z); --z) {
96  emit Updated(x, y, z);
97  const ushort transparent =
98  shred->GetBlock(x_in, y_in, z)->Transparent();
99  if ( BLOCK_TRANSPARENT == transparent ) {
100  --light_lev;
101  } else if ( BLOCK_OPAQUE == transparent ) {
102  break;
103  }
104  }
105  UpShine(x-1, y, z);
106  UpShine(x+1, y, z);
107  UpShine(x, y-1, z);
108  UpShine(x, y+1, z);
109 }

158 void World::ReEnlightenMove(const int dir) {
159  disconnect(this, SIGNAL(Updated(ushort, ushort, ushort)), 0, 0);
160  disconnect(this, SIGNAL(UpdatedAround(ushort, ushort, ushort, ushort)),
