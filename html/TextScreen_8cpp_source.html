64 void Screen::ControlPlayer(const int ch) {
65  CleanFileToShow();
66  // Q, ctrl-c, ctrl-d, ctrl-q, ctrl-x
67  // TODO: ctrl-z (to background) support
68  if ( 'Q'==ch || 3==ch || 4==ch || 17==ch || 24==ch ) {
69  emit ExitReceived();
70  return;
71  } // else:
72  if ( ch>='a' && ch<='z' ) { // actions with inventory
73  InventoryAction(ch-'a');
74  return;
75  } // else:
76  switch ( ch ) { // interactions with world
77  case '8': MovePlayer(NORTH); break;
78  case '2': MovePlayer(SOUTH); break;
79  case '6': MovePlayer(EAST); break;
80  case '4': MovePlayer(WEST); break;
81  case '5': player->Move(DOWN); break;
82  case '7': MovePlayerDiag(NORTH, WEST); break;
83  case '9': MovePlayerDiag(NORTH, EAST); break;
84  case '1': MovePlayerDiag(SOUTH, WEST); break;
85  case '3': MovePlayerDiag(SOUTH, EAST); break;
86  case ' ': player->Jump(); break;
87 
88  case '>': player->SetDir(World::TurnRight(player->GetDir())); break;
89  case '<': player->SetDir(World::TurnLeft (player->GetDir())); break;
90  case 'A': player->SetDir(DOWN); break;
91  case 'Z': player->SetDir(UP); break;
92 
93  case 'I': player->Backpack(); break;
94  case 8: player->Damage(); break;
95  case 13:
96  case '\n': player->Use(); break;
97  case '?': player->Examine(); break;
98  case '~': player->Inscribe(); break;
99  case 27: /* esc */ player->StopUseAll(); break;
100 
101  case 'B': SetActionMode(ACTION_BUILD); break;
102  case 'C': SetActionMode(ACTION_CRAFT); break;
103  case 'D':
104  case 'T': SetActionMode(ACTION_THROW); break;
105  case 'N': SetActionMode(ACTION_INSCRIBE); break;
106  case 'G':
107  case 'O': SetActionMode(ACTION_OBTAIN); break;
108  case 'E':
109  case 'U': SetActionMode(ACTION_USE); break;
110  case 'F':
111  case 'W': SetActionMode(ACTION_WIELD); break;
112  case 'S':
113  if ( player->PlayerInventory() ) {
114  player->PlayerInventory()->Shake();
115  }
116  break;
117  case 'H':
118  DisplayFile(QString("help_%1/help.txt").arg(locale.left(2)));
119  break;
120 
121  case '-': shiftFocus = -!shiftFocus; break; // move focus down
122  case '+': shiftFocus = !shiftFocus; break; // move focus up
123 
124  case '!': player->SetCreativeMode( not player->GetCreativeMode() ); break;
125  case ':':
126  case '/': PassString(command); // no break
127  case '.': ProcessCommand(command); break;
128 
129  default: Notify(tr("Unknown key. Press 'H' for help."));
130  }
