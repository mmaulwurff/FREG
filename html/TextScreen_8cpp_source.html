67 void Screen::ControlPlayer(const int ch) {
68  CleanFileToShow();
69  // Q, ctrl-c, ctrl-d, ctrl-q, ctrl-x
70  // TODO: ctrl-z (to background) support
71  if ( 'Q'==ch || 3==ch || 4==ch || 17==ch || 24==ch ) {
72  emit ExitReceived();
73  return;
74  } // else:
75  if ( ch>='a' && ch<='z' ) { // actions with inventory
76  InventoryAction(ch-'a');
77  return;
78  } // else:
79  switch ( ch ) { // interactions with world
80  case '8': MovePlayer(NORTH); break;
81  case '2': MovePlayer(SOUTH); break;
82  case '6': MovePlayer(EAST); break;
83  case '4': MovePlayer(WEST); break;
84  case '5': player->Move(DOWN); break;
85  case '7': MovePlayerDiag(NORTH, WEST); break;
86  case '9': MovePlayerDiag(NORTH, EAST); break;
87  case '1': MovePlayerDiag(SOUTH, WEST); break;
88  case '3': MovePlayerDiag(SOUTH, EAST); break;
89  case ' ': player->Jump(); break;
90  case '=': player->Move(); break;
91 
92  case '>': player->SetDir(World::TurnRight(player->GetDir())); break;
93  case '<': player->SetDir(World::TurnLeft (player->GetDir())); break;
94  case 'A': player->SetDir(DOWN); break;
95  case 'Z': player->SetDir(UP); break;
96 
97  case 'I': player->Backpack(); break;
98  case 8: player->Damage(); break;
99  case 13:
100  case '\n': player->Use(); break;
101  case '?': player->Examine(); break;
102  case '~': player->Inscribe(); break;
103  case 27: /* esc */ player->StopUseAll(); break;
104 
105  case 'B': SetActionMode(ACTION_BUILD); break;
106  case 'C': SetActionMode(ACTION_CRAFT); break;
107  case 'D':
108  case 'T': SetActionMode(ACTION_THROW); break;
109  case 'E': SetActionMode(ACTION_EAT); break;
110  case 'F': SetActionMode(ACTION_TAKEOFF); break;
111  case 'N': SetActionMode(ACTION_INSCRIBE); break;
112  case 'G':
113  case 'O': SetActionMode(ACTION_OBTAIN); break;
114  case 'U': SetActionMode(ACTION_USE); break;
115  case 'W': SetActionMode(ACTION_WIELD); break;
116  case 'S':
117  if ( player->PlayerInventory() ) {
118  player->PlayerInventory()->Shake();
119  }
120  break;
121  case 'H':
122  DisplayFile(QString("help_%1/help.txt").arg(locale.left(2)));
123  break;
124  case 'R': // switch active hand
125  if ( not player->GetCreativeMode() ) {
126  player->SetActiveHand(not player->IsRightActiveHand());
127  Notify(tr("Now %1 hand is active.").
128  arg(player->IsRightActiveHand() ?
129  tr("right") : tr("left")));
130  }
131  break;
132 
133  case '-': shiftFocus = -!shiftFocus; break; // move focus down
134  case '+': shiftFocus = !shiftFocus; break; // move focus up
135 
136  case '!': player->SetCreativeMode( not player->GetCreativeMode() ); break;
137  case ':':
138  case '/': PassString(command); // no break
139  case '.': ProcessCommand(command); break;
140 
141  default: Notify(tr("Unknown key. Press 'H' for help."));
142  }
143 } // void Screen::ControlPlayer(int ch)
