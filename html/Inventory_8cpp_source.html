213 bool Inventory::MiniCraft(const ushort num) {

215  ReceiveSignal(QObject::tr("Nothing here."));

217  } // else:

219  Number(num), BlockManager::MakeId(GetInvKind(num),GetInvSub(num)));
220  if ( crafted != nullptr ) {

222  Block * const to_delete = ShowBlock(num);
