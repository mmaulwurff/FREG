374  switch ( TimeOfDay() ) {
375  case END_OF_NIGHT:
376  ReEnlightenTime();

387  emit Notify(tr("It's night now."));

391  //emit ExitReceived(); //close all after 1 turn
392  Unlock();
393 } //World::PhysEvents

492 bool World::CanMove(
493  const ushort i, const ushort j, const ushort k,
494  const ushort newi, const ushort newj, const ushort newk,
495  const quint8 dir)
496 {
497  Block * const block=GetBlock(i, j, k);
498  if ( NOT_MOVABLE==block->Movable() ) {
499  return false;
500  }
501  Block * block_to=GetBlock(newi, newj, newk);
502  if ( ENVIRONMENT==block->Movable() && *block==*block_to ) {
503  return false;
504  }
505  switch ( block_to->BeforePush(dir, block) ) {
506  case DESTROY:
507  DeleteBlock(block_to);
508  //BeforePush could change block_to:
509  PutBlock((block_to=Normal(AIR)), newi, newj, newk);
510  break;
511  case MOVE_SELF:
512  block_to=GetBlock(newi, newj, newk);
513  break;
514  case JUMP:
515  if ( DOWN!=dir && UP!=dir ) {
516  Jump(i, j, k, dir);
517  return false;
518  }
519  break;
520  case MOVE_UP:
521  if ( DOWN!=dir ) {
522  Move(i, j, k, UP);
523  return false;
524  }
525  break;
526  case DAMAGE:
527  block->ReceiveSignal(tr("Ouch!"));
528  Damage(i, j, k,
529  block_to->DamageLevel(),
530  block_to->DamageKind());
531  return false;
532  break;
533  }
534  return (( ENVIRONMENT==block_to->Movable() ) ||
535  Move(newi, newj, newk, dir));
536 } //World::CanMove
