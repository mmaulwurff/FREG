508 bool World::CanMove(const ushort x, const ushort y, const ushort z,
509  const ushort newx, const ushort newy, const ushort newz,
510  const quint8 dir)
511 {
512  bool move_flag;
513  Block * const block = GetBlock(x, y, z);
514  Block * block_to = GetBlock(newx, newy, newz);
515  if ( ENVIRONMENT == block->PushResult(NOWHERE) ) {
516  const int target_push = block_to->PushResult(NOWHERE);
517  move_flag = (*block != *block_to)
518  && ( MOVABLE == target_push || ENVIRONMENT == target_push );
519  } else {
520  block_to->Push(dir, block);
521  block_to = GetBlock(newx, newy, newz);
522  switch ( block_to->PushResult(dir) ) {
523  default:
524  case NOT_MOVABLE: move_flag = false; break;
525  case ENVIRONMENT: move_flag = true; break;
526  case JUMP:
527  if ( dir > DOWN ) { // not UP and not DOWN
528  Jump(x, y, z, dir);
529  }
530  move_flag = false;
531  break;
532  case MOVE_UP:
533  if ( dir > DOWN ) { // not UP and not DOWN
534  Move(x, y, z, UP);
535  }
536  move_flag = false;
537  break;
538  case MOVABLE:
539  move_flag = ( (block->Weight() > block_to->Weight()) &&
540  Move(newx, newy, newz, dir) );
541  break;
542  }
543  }
544  const Active * active;
545  return ( move_flag &&
546  (DOWN==dir || !block->Weight() ||
547  !( (active=block->ActiveBlock()) &&
548  active->IsFalling() &&
549  AIR==GetBlock(x, y, z-1)->Sub() &&
550  AIR==GetBlock(newx, newy, newz-1)->Sub() )) );
551 } // bool World::CanMove(ushort x, y, z, newx, newy, newz, quint8 dir)
