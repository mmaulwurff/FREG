40 QString Bucket::FullName() const {
41  QString name;
42  switch (GetInvSub(0)) {
43  case AIR: return QObject::tr("Empty bucket"); break;
44  case WATER: name = QObject::tr("Bucket with water"); break;
45  case STONE: name = QObject::tr("Bucket with lava"); break;

48  GetInvSub(0));

52  arg(Number(0)).arg(MAX_STACK_SIZE));
